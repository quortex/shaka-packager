93 bool DetermineTextFileCodec(const std::string& file, std::string* out) {
94  CHECK(out);
95 
96  std::string content;
97  if (!File::ReadFileToString(file.c_str(), &content)) {
98  LOG(ERROR) << "Failed to open file " << file
99  << " to determine file format.";
100  return false;
101  }
102 
103  const uint8_t* content_data =
104  reinterpret_cast<const uint8_t*>(content.data());
105  MediaContainerName container_name =
106  DetermineContainer(content_data, content.size());
107 
108  if (container_name == CONTAINER_WEBVTT) {
109  *out = "wvtt";
110  return true;
111  }
112 
113  if (container_name == CONTAINER_TTML) {
114  *out = "ttml";
115  return true;
116  }
117 
118  return false;
119 }

121 MediaContainerName GetOutputFormat(const StreamDescriptor& descriptor) {
122  if (!descriptor.output_format.empty()) {
123  MediaContainerName format =
124  DetermineContainerFromFormatName(descriptor.output_format);
125  if (format == CONTAINER_UNKNOWN) {
126  LOG(ERROR) << "Unable to determine output format from '"
127  << descriptor.output_format << "'.";
128  }
129  return format;
130  }
131 
132  base::Optional<MediaContainerName> format_from_output;
133  base::Optional<MediaContainerName> format_from_segment;
134  if (!descriptor.output.empty()) {
135  format_from_output = DetermineContainerFromFileName(descriptor.output);
136  if (format_from_output.value() == CONTAINER_UNKNOWN) {
137  LOG(ERROR) << "Unable to determine output format from '"
138  << descriptor.output << "'.";
139  }
140  }
141  if (!descriptor.segment_template.empty()) {
142  format_from_segment =
143  DetermineContainerFromFileName(descriptor.segment_template);
144  if (format_from_segment.value() == CONTAINER_UNKNOWN) {
145  LOG(ERROR) << "Unable to determine output format from '"
146  << descriptor.segment_template << "'.";
147  }
148  }
149 
150  if (format_from_output && format_from_segment) {
151  if (format_from_output.value() != format_from_segment.value()) {
152  LOG(ERROR) << "Output format determined from '" << descriptor.output
153  << "' differs from output format determined from '"
154  << descriptor.segment_template << "'.";
155  return CONTAINER_UNKNOWN;
156  }
157  }
158 
159  if (format_from_output)
160  return format_from_output.value();
161  if (format_from_segment)
162  return format_from_segment.value();
163  return CONTAINER_UNKNOWN;
164 }

166 MediaContainerName GetTextOutputCodec(const StreamDescriptor& descriptor) {
167  const auto output_container = GetOutputFormat(descriptor);
168  if (output_container != CONTAINER_MOV)
169  return output_container;
170 
171  const auto input_container = DetermineContainerFromFileName(descriptor.input);
172  if (base::EqualsCaseInsensitiveASCII(descriptor.output_format, "vtt+mp4") ||
173  base::EqualsCaseInsensitiveASCII(descriptor.output_format,
174  "webvtt+mp4")) {
175  return CONTAINER_WEBVTT;
176  } else if (!base::EqualsCaseInsensitiveASCII(descriptor.output_format,
177  "ttml+mp4") &&
178  input_container == CONTAINER_WEBVTT) {
179  // With WebVTT input, default to WebVTT output.
180  return CONTAINER_WEBVTT;
181  } else {
182  // Otherwise default to TTML since it has more features.
183  return CONTAINER_TTML;
184  }
185 }

188  if (stream.stream_selector == "text")
189  return true;
190  if (base::EqualsCaseInsensitiveASCII(stream.output_format, "vtt+mp4") ||
191  base::EqualsCaseInsensitiveASCII(stream.output_format, "webvtt+mp4") ||

195 
196  auto output_format = GetOutputFormat(stream);
197  return output_format == CONTAINER_WEBVTT || output_format == CONTAINER_TTML;

200 Status ValidateStreamDescriptor(bool dump_stream_info,
201  const StreamDescriptor& stream) {
202  if (stream.input.empty()) {
203  return Status(error::INVALID_ARGUMENT, "Stream input not specified.");
204  }
205 
206  // The only time a stream can have no outputs, is when dump stream info is
207  // set.
208  if (dump_stream_info && stream.output.empty() &&
209  stream.segment_template.empty()) {
210  return Status::OK;
211  }
212 
213  if (stream.output.empty() && stream.segment_template.empty()) {
214  return Status(error::INVALID_ARGUMENT,
215  "Streams must specify 'output' or 'segment template'.");
216  }
217 
218  // Whenever there is output, a stream must be selected.
219  if (stream.stream_selector.empty()) {
220  return Status(error::INVALID_ARGUMENT,
221  "Stream stream_selector not specified.");
222  }
223 
224  // If a segment template is provided, it must be valid.
225  if (stream.segment_template.length()) {
226  RETURN_IF_ERROR(ValidateSegmentTemplate(stream.segment_template));
227  }
228 
229  // There are some specifics that must be checked based on which format
230  // we are writing to.
231  const MediaContainerName output_format = GetOutputFormat(stream);
232 
233  if (output_format == CONTAINER_UNKNOWN) {
234  return Status(error::INVALID_ARGUMENT, "Unsupported output format.");
235  }
236  if (output_format == MediaContainerName::CONTAINER_MPEG2TS) {
237  if (stream.segment_template.empty()) {
238  return Status(
239  error::INVALID_ARGUMENT,
240  "Please specify 'segment_template'. Single file TS output is "
241  "not supported.");
242  }
243 
244  // Right now the init segment is saved in |output| for multi-segment
245  // content. However, for TS all segments must be self-initializing so
246  // there cannot be an init segment.
247  if (stream.output.length()) {
248  return Status(error::INVALID_ARGUMENT,
249  "All TS segments must be self-initializing. Stream "
250  "descriptors 'output' or 'init_segment' are not allowed.");
251  }
252  } else if (output_format == CONTAINER_WEBVTT ||
253  output_format == CONTAINER_TTML ||
254  output_format == CONTAINER_AAC || output_format == CONTAINER_MP3 ||
255  output_format == CONTAINER_AC3 ||
256  output_format == CONTAINER_EAC3) {
257  // There is no need for an init segment when outputting because there is no
258  // initialization data.
259  if (stream.segment_template.length() && stream.output.length()) {
260  return Status(
261  error::INVALID_ARGUMENT,
262  "Segmented subtitles or PackedAudio output cannot have an init "
263  "segment. Do not specify stream descriptors 'output' or "
264  "'init_segment' when using 'segment_template'.");
265  }
266  } else {
267  // For any other format, if there is a segment template, there must be an
268  // init segment provided.
269  if (stream.segment_template.length() && stream.output.empty()) {
270  return Status(error::INVALID_ARGUMENT,
271  "Please specify 'init_segment'. All non-TS multi-segment "
272  "content must provide an init segment.");
273  }
274  }
275 
276  if (stream.output.find('$') != std::string::npos) {
277  if (output_format == CONTAINER_WEBVTT) {
278  return Status(
279  error::UNIMPLEMENTED,
280  "WebVTT output with one file per Representation per Period "
281  "is not supported yet. Please use fMP4 instead. If that needs to be "
282  "supported, please file a feature request on GitHub.");
283  }
284  // "$" is only allowed if the output file name is a template, which is
285  // used to support one file per Representation per Period when there are
286  // Ad Cues.
287  RETURN_IF_ERROR(ValidateSegmentTemplate(stream.output));
288  }
289 
290  return Status::OK;
291 }

293 Status ValidateParams(const PackagingParams& packaging_params,
294  const std::vector<StreamDescriptor>& stream_descriptors) {
295  if (!packaging_params.chunking_params.segment_sap_aligned &&
296  packaging_params.chunking_params.subsegment_sap_aligned) {
297  return Status(error::INVALID_ARGUMENT,
298  "Setting segment_sap_aligned to false but "
299  "subsegment_sap_aligned to true is not allowed.");
300  }
301 
302  if (stream_descriptors.empty()) {
303  return Status(error::INVALID_ARGUMENT,
304  "Stream descriptors cannot be empty.");
305  }
306 
307  // On demand profile generates single file segment while live profile
308  // generates multiple segments specified using segment template.
309  const bool on_demand_dash_profile =
310  stream_descriptors.begin()->segment_template.empty();
311  std::set<std::string> outputs;
312  std::set<std::string> segment_templates;
313  for (const auto& descriptor : stream_descriptors) {
314  if (on_demand_dash_profile != descriptor.segment_template.empty()) {
315  return Status(error::INVALID_ARGUMENT,
316  "Inconsistent stream descriptor specification: "
317  "segment_template should be specified for none or all "
318  "stream descriptors.");
319  }
320 
321  RETURN_IF_ERROR(ValidateStreamDescriptor(
322  packaging_params.test_params.dump_stream_info, descriptor));
323 
324  if (base::StartsWith(descriptor.input, "udp://",
325  base::CompareCase::SENSITIVE)) {
326  const HlsParams& hls_params = packaging_params.hls_params;
327  if (!hls_params.master_playlist_output.empty() &&
328  hls_params.playlist_type == HlsPlaylistType::kVod) {
329  LOG(WARNING)
330  << "Seeing UDP input with HLS Playlist Type set to VOD. The "
331  "playlists will only be generated when UDP socket is closed. "
332  "If you want to do live packaging, --hls_playlist_type needs to "
333  "be set to LIVE.";
334  }
335  // Skip the check for DASH as DASH defaults to 'dynamic' MPD when segment
336  // template is provided.
337  }
338 
339  if (!descriptor.output.empty()) {
340  if (outputs.find(descriptor.output) != outputs.end()) {
341  return Status(
342  error::INVALID_ARGUMENT,
343  "Seeing duplicated outputs '" + descriptor.output +
344  "' in stream descriptors. Every output must be unique.");
345  }
346  outputs.insert(descriptor.output);
347  }
348  if (!descriptor.segment_template.empty()) {
349  if (segment_templates.find(descriptor.segment_template) !=
350  segment_templates.end()) {
351  return Status(error::INVALID_ARGUMENT,
352  "Seeing duplicated segment templates '" +
353  descriptor.segment_template +
354  "' in stream descriptors. Every segment template "
355  "must be unique.");
356  }
357  segment_templates.insert(descriptor.segment_template);
358  }
359  }
360 
361  if (packaging_params.output_media_info && !on_demand_dash_profile) {
362  // TODO(rkuroiwa, kqyang): Support partial media info dump for live.
363  return Status(error::UNIMPLEMENTED,
364  "--output_media_info is only supported for on-demand profile "
365  "(not using segment_template).");
366  }
367 
368  if (on_demand_dash_profile &&
369  !packaging_params.mpd_params.mpd_output.empty() &&
370  !packaging_params.mp4_output_params.generate_sidx_in_media_segments &&
371  !packaging_params.mpd_params.use_segment_list) {
372  return Status(error::UNIMPLEMENTED,
373  "--generate_sidx_in_media_segments is required for DASH "
374  "on-demand profile (not using segment_template or segment list).");
375  }
376 
377  if (packaging_params.chunking_params.low_latency_dash_mode &&
378  packaging_params.chunking_params.subsegment_duration_in_seconds) {
379  // Low latency streaming requires data to be shipped as chunks,
380  // the smallest unit of video. Right now, each chunk contains
381  // one frame. Therefore, in low latency mode,
382  // a user specified --fragment_duration is irrelevant.
383  // TODO(caitlinocallaghan): Add a feature for users to specify the number
384  // of desired frames per chunk.
385  return Status(error::INVALID_ARGUMENT,
386  "--fragment_duration cannot be set "
387  "if --low_latency_dash_mode is enabled.");
388  }
389 
390  if (packaging_params.mpd_params.low_latency_dash_mode &&
391  packaging_params.mpd_params.utc_timings.empty()) {
392  // Low latency DASH MPD requires a UTC Timing value
393  return Status(error::INVALID_ARGUMENT,
394  "--utc_timings must be be set "
395  "if --low_latency_dash_mode is enabled.");
396  }
397 
398  return Status::OK;
399 }

401 bool StreamDescriptorCompareFn(const StreamDescriptor& a,
402  const StreamDescriptor& b) {
403  // This function is used by std::sort() to sort the stream descriptors.
404  // Note that std::sort() need a comparator that return true iff the first
405  // argument is strictly lower than the second one. That is: must return false
406  // when they are equal. The requirement is enforced in gcc/g++ but not in
407  // clang.
408  if (a.input == b.input) {
409  if (a.stream_selector == b.stream_selector) {
410  // The MPD notifier requires that the main track comes first, so make
411  // sure that happens.
412  return a.trick_play_factor < b.trick_play_factor;
413  }
414  return a.stream_selector < b.stream_selector;
415  }
416 
417  return a.input < b.input;
418 }

427 bool StreamInfoToTextMediaInfo(const StreamDescriptor& stream_descriptor,
428  MediaInfo* text_media_info) {
429  std::string codec;
430  if (!DetermineTextFileCodec(stream_descriptor.input, &codec)) {
431  LOG(ERROR) << "Failed to determine the text file format for "
432  << stream_descriptor.input;
433  return false;
434  }
435 
436  MediaInfo::TextInfo* text_info = text_media_info->mutable_text_info();
437  text_info->set_codec(codec);
438 
439  const std::string& language = stream_descriptor.language;
440  if (!language.empty()) {
441  text_info->set_language(language);
442  }
443 
444  text_media_info->set_media_file_name(stream_descriptor.output);
445  text_media_info->set_container_type(MediaInfo::CONTAINER_TEXT);
446 
447  if (stream_descriptor.bandwidth != 0) {
448  text_media_info->set_bandwidth(stream_descriptor.bandwidth);
449  } else {
450  // Text files are usually small and since the input is one file; there's no
451  // way for the player to do ranged requests. So set this value to something
452  // reasonable.
453  const int kDefaultTextBandwidth = 256;
454  text_media_info->set_bandwidth(kDefaultTextBandwidth);
455  }
456 
457  if (!stream_descriptor.dash_roles.empty()) {
458  for (const auto& dash_role : stream_descriptor.dash_roles) {
459  text_media_info->add_dash_roles(dash_role);
460  }
461  }
462 
463  return true;
464 }

490 std::shared_ptr<MediaHandler> CreateEncryptionHandler(
491  const PackagingParams& packaging_params,
492  const StreamDescriptor& stream,
493  KeySource* key_source) {
494  if (stream.skip_encryption) {
495  return nullptr;
496  }
497 
498  if (!key_source) {
499  return nullptr;
500  }
501 
502  // Make a copy so that we can modify it for this specific stream.
503  EncryptionParams encryption_params = packaging_params.encryption_params;
504 
505  // Use Sample AES in MPEG2TS.
506  // TODO(kqyang): Consider adding a new flag to enable Sample AES as we
507  // will support CENC in TS in the future.
508  if (GetOutputFormat(stream) == CONTAINER_MPEG2TS ||
509  GetOutputFormat(stream) == CONTAINER_AAC ||
510  GetOutputFormat(stream) == CONTAINER_AC3 ||
511  GetOutputFormat(stream) == CONTAINER_EAC3) {
512  VLOG(1) << "Use Apple Sample AES encryption for MPEG2TS or Packed Audio.";
513  encryption_params.protection_scheme = kAppleSampleAesProtectionScheme;
514  }
515 
516  if (!stream.drm_label.empty()) {
517  const std::string& drm_label = stream.drm_label;
518  encryption_params.stream_label_func =
519  [drm_label](const EncryptionParams::EncryptedStreamAttributes&) {
520  return drm_label;
521  };
522  } else if (!encryption_params.stream_label_func) {
523  const int kDefaultMaxSdPixels = 768 * 576;
524  const int kDefaultMaxHdPixels = 1920 * 1080;
525  const int kDefaultMaxUhd1Pixels = 4096 * 2160;
526  encryption_params.stream_label_func = std::bind(
527  &Packager::DefaultStreamLabelFunction, kDefaultMaxSdPixels,
528  kDefaultMaxHdPixels, kDefaultMaxUhd1Pixels, std::placeholders::_1);
529  }
530 
531  return std::make_shared<EncryptionHandler>(encryption_params, key_source);
532 }

542 Status CreateTtmlJobs(
543  const std::vector<std::reference_wrapper<const StreamDescriptor>>& streams,
544  const PackagingParams& packaging_params,
545  SyncPointQueue* sync_points,
546  MuxerFactory* muxer_factory,
547  MpdNotifier* mpd_notifier,
548  JobManager* job_manager) {
549  DCHECK(job_manager);
550  for (const StreamDescriptor& stream : streams) {
551  // Check input to ensure that output is possible.
552  if (!packaging_params.hls_params.master_playlist_output.empty() &&
553  !stream.dash_only) {
554  return Status(error::INVALID_ARGUMENT,
555  "HLS does not support TTML in xml format.");
556  }
557 
558  if (!stream.segment_template.empty()) {
559  return Status(error::INVALID_ARGUMENT,
560  "Segmented TTML is not supported.");
561  }
562 
563  if (GetOutputFormat(stream) != CONTAINER_TTML) {
564  return Status(error::INVALID_ARGUMENT,
565  "Converting TTML to other formats is not supported");
566  }
567 
568  if (!stream.output.empty()) {
569  if (!File::Copy(stream.input.c_str(), stream.output.c_str())) {
570  std::string error;
571  base::StringAppendF(
572  &error, "Failed to copy the input file (%s) to output file (%s).",
573  stream.input.c_str(), stream.output.c_str());
574  return Status(error::FILE_FAILURE, error);
575  }
576 
577  MediaInfo text_media_info;
578  if (!StreamInfoToTextMediaInfo(stream, &text_media_info)) {
579  return Status(error::INVALID_ARGUMENT,
580  "Could not create media info for stream.");
581  }
582 
583  // If we are outputting to MPD, just add the input to the outputted
584  // manifest.
585  if (mpd_notifier) {
586  uint32_t unused;
587  if (mpd_notifier->NotifyNewContainer(text_media_info, &unused)) {
588  mpd_notifier->Flush();
589  } else {
590  return Status(error::PARSER_FAILURE,
591  "Failed to process text file " + stream.input);
592  }
593  }
594 
595  if (packaging_params.output_media_info) {
596  VodMediaInfoDumpMuxerListener::WriteMediaInfoToFile(
597  text_media_info, stream.output + kMediaInfoSuffix);
598  }
599  }
600  }
601 
602  return Status::OK;
603 }

610  MuxerListenerFactory* muxer_listener_factory,
611  MuxerFactory* muxer_factory,
612  JobManager* job_manager) {
613  DCHECK(muxer_listener_factory);
614  DCHECK(muxer_factory);
615  DCHECK(job_manager);
616  // Store all the demuxers in a map so that we can look up a stream's demuxer.
617  // This is step one in making this part of the pipeline less dependant on
618  // order.
619  std::map<std::string, std::shared_ptr<Demuxer>> sources;
620  std::map<std::string, std::shared_ptr<MediaHandler>> cue_aligners;
621 
622  for (const StreamDescriptor& stream : streams) {
623  bool seen_input_before = sources.find(stream.input) != sources.end();
624  if (seen_input_before) {
625  continue;
626  }
627 
628  RETURN_IF_ERROR(
629  CreateDemuxer(stream, packaging_params, &sources[stream.input]));
630  cue_aligners[stream.input] =
631  sync_points ? std::make_shared<CueAlignmentHandler>(sync_points)
632  : nullptr;
633  }
634 
635  for (auto& source : sources) {
636  job_manager->Add("RemuxJob", source.second);
637  }
638 
639  // Replicators are shared among all streams with the same input and stream
640  // selector.
641  std::shared_ptr<MediaHandler> replicator;
642 
643  std::string previous_input;
644  std::string previous_selector;
645 
646  for (const StreamDescriptor& stream : streams) {
647  // Get the demuxer for this stream.
648  auto& demuxer = sources[stream.input];
649  auto& cue_aligner = cue_aligners[stream.input];
650 
651  const bool new_input_file = stream.input != previous_input;
652  const bool new_stream =
653  new_input_file || previous_selector != stream.stream_selector;
654  const bool is_text = IsTextStream(stream);
655  previous_input = stream.input;
656  previous_selector = stream.stream_selector;
657 
658  // If the stream has no output, then there is no reason setting-up the rest
659  // of the pipeline.
660  if (stream.output.empty() && stream.segment_template.empty()) {
661  continue;
662  }
663 
664  // Just because it is a different stream descriptor does not mean it is a
665  // new stream. Multiple stream descriptors may have the same stream but
666  // only differ by trick play factor.
667  if (new_stream) {
668  if (!stream.language.empty()) {
669  demuxer->SetLanguageOverride(stream.stream_selector, stream.language);
670  }
671 
672  std::vector<std::shared_ptr<MediaHandler>> handlers;
673  if (is_text) {
674  handlers.emplace_back(
675  std::make_shared<TextPadder>(kDefaultTextZeroBiasMs));
676  }
677  if (sync_points) {
678  handlers.emplace_back(cue_aligner);
679  }
680  if (!is_text) {
681  handlers.emplace_back(std::make_shared<ChunkingHandler>(
682  packaging_params.chunking_params));
683  handlers.emplace_back(CreateEncryptionHandler(packaging_params, stream,
684  encryption_key_source));
685  }
686 
687  replicator = std::make_shared<Replicator>();
688  handlers.emplace_back(replicator);
689 
690  RETURN_IF_ERROR(MediaHandler::Chain(handlers));
691  RETURN_IF_ERROR(demuxer->SetHandler(stream.stream_selector, handlers[0]));
692  }
693 
694  // Create the muxer (output) for this track.
695  const auto output_format = GetOutputFormat(stream);
696  std::shared_ptr<Muxer> muxer =
697  muxer_factory->CreateMuxer(output_format, stream);
698  if (!muxer) {
699  return Status(error::INVALID_ARGUMENT, "Failed to create muxer for " +
700  stream.input + ":" +
701  stream.stream_selector);
702  }
703 
704  std::unique_ptr<MuxerListener> muxer_listener =
705  muxer_listener_factory->CreateListener(ToMuxerListenerData(stream));
706  muxer->SetMuxerListener(std::move(muxer_listener));
707 
708  std::vector<std::shared_ptr<MediaHandler>> handlers;
709  handlers.emplace_back(replicator);
710 
711  // Trick play is optional.
712  if (stream.trick_play_factor) {
713  handlers.emplace_back(
714  std::make_shared<TrickPlayHandler>(stream.trick_play_factor));
715  }
716 
717  if (stream.cc_index >= 0) {
718  handlers.emplace_back(
719  std::make_shared<CcStreamFilter>(stream.language, stream.cc_index));
720  }
721 
722  if (is_text &&
723  (!stream.segment_template.empty() || output_format == CONTAINER_MOV)) {
724  handlers.emplace_back(
725  CreateTextChunker(packaging_params.chunking_params));
726  }
727 
728  if (is_text && output_format == CONTAINER_MOV) {
729  const auto output_codec = GetTextOutputCodec(stream);
730  if (output_codec == CONTAINER_WEBVTT) {
731  handlers.emplace_back(std::make_shared<WebVttToMp4Handler>());
732  } else if (output_codec == CONTAINER_TTML) {
733  handlers.emplace_back(std::make_shared<ttml::TtmlToMp4Handler>());
734  }
735  }
736 
737  handlers.emplace_back(muxer);
738  RETURN_IF_ERROR(MediaHandler::Chain(handlers));
739  }
740 
741  return Status::OK;
742 }

744 Status CreateAllJobs(const std::vector<StreamDescriptor>& stream_descriptors,
745  const PackagingParams& packaging_params,
746  MpdNotifier* mpd_notifier,
747  KeySource* encryption_key_source,
748  SyncPointQueue* sync_points,
749  MuxerListenerFactory* muxer_listener_factory,
750  MuxerFactory* muxer_factory,
751  JobManager* job_manager) {
752  DCHECK(muxer_factory);
753  DCHECK(muxer_listener_factory);
754  DCHECK(job_manager);
755 
756  // Group all streams based on which pipeline they will use.
757  std::vector<std::reference_wrapper<const StreamDescriptor>> ttml_streams;
758  std::vector<std::reference_wrapper<const StreamDescriptor>>
759  audio_video_streams;
760 
761  bool has_transport_audio_video_streams = false;
762  bool has_non_transport_audio_video_streams = false;
763 
764  for (const StreamDescriptor& stream : stream_descriptors) {
765  const auto input_container = DetermineContainerFromFileName(stream.input);
766  const auto output_format = GetOutputFormat(stream);
767  if (input_container == CONTAINER_TTML) {
768  ttml_streams.push_back(stream);
769  } else {
770  audio_video_streams.push_back(stream);
771  switch (output_format) {
772  case CONTAINER_MPEG2TS:
773  case CONTAINER_AAC:
774  case CONTAINER_MP3:
775  case CONTAINER_AC3:
776  case CONTAINER_EAC3:
777  has_transport_audio_video_streams = true;
778  break;
779  case CONTAINER_TTML:
780  case CONTAINER_WEBVTT:
781  break;
782  default:
783  has_non_transport_audio_video_streams = true;
784  break;
785  }
786  }
787  }
788 
789  // Audio/Video streams need to be in sorted order so that demuxers and trick
790  // play handlers get setup correctly.
791  std::sort(audio_video_streams.begin(), audio_video_streams.end(),
792  media::StreamDescriptorCompareFn);
793 
794  if (packaging_params.transport_stream_timestamp_offset_ms > 0) {
795  if (has_transport_audio_video_streams &&
796  has_non_transport_audio_video_streams) {
797  LOG(WARNING) << "There may be problems mixing transport streams and "
798  "non-transport streams. For example, the subtitles may "
799  "be out of sync with non-transport streams.";
800  } else if (has_non_transport_audio_video_streams) {
801  // Don't insert the X-TIMESTAMP-MAP in WebVTT if there is no transport
802  // stream.
803  muxer_factory->SetTsStreamOffset(0);
804  }
805  }
806 
807  RETURN_IF_ERROR(CreateTtmlJobs(ttml_streams, packaging_params, sync_points,
808  muxer_factory, mpd_notifier, job_manager));
809  RETURN_IF_ERROR(CreateAudioVideoJobs(
810  audio_video_streams, packaging_params, encryption_key_source, sync_points,
811  muxer_listener_factory, muxer_factory, job_manager));
812 
813  // Initialize processing graph.
814  return job_manager->InitializeJobs();
815 }
