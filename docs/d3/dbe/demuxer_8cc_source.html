39 std::string GetStreamLabel(size_t stream_index) {
40  switch (stream_index) {
41  case kBaseVideoOutputStreamIndex:
42  return "video";
43  case kBaseAudioOutputStreamIndex:
44  return "audio";
45  case kBaseTextOutputStreamIndex:
46  return "text";
47  default:
48  return base::SizeTToString(stream_index);
49  }
50 }

52 bool GetStreamIndex(const std::string& stream_label, size_t* stream_index) {
53  DCHECK(stream_index);
54  if (stream_label == "video") {
55  *stream_index = kBaseVideoOutputStreamIndex;
56  } else if (stream_label == "audio") {
57  *stream_index = kBaseAudioOutputStreamIndex;
58  } else if (stream_label == "text") {
59  *stream_index = kBaseTextOutputStreamIndex;
60  } else {
61  // Expect stream_label to be a zero based stream id.
62  if (!base::StringToSizeT(stream_label, stream_index)) {
63  LOG(ERROR) << "Invalid argument --stream=" << stream_label << "; "
64  << "should be 'audio', 'video', 'text', or a number";
65  return false;
66  }
67  }
68  return true;
69 }

88 Status Demuxer::Run() {
89  LOG(INFO) << "Demuxer::Run() on file '" << file_name_ << "'.";
90  Status status = InitializeParser();
91  // ParserInitEvent callback is called after a few calls to Parse(), which sets
92  // up the streams. Only after that, we can verify the outputs below.
93  while (!all_streams_ready_ && status.ok())
94  status.Update(Parse());
95  // If no output is defined, then return success after receiving all stream
96  // info.
97  if (all_streams_ready_ && output_handlers().empty())
98  return Status::OK;
99  if (!init_event_status_.ok())
100  return init_event_status_;
101  if (!status.ok())
102  return status;
103  // Check if all specified outputs exists.
104  for (const auto& pair : output_handlers()) {
105  if (std::find(stream_indexes_.begin(), stream_indexes_.end(), pair.first) ==
106  stream_indexes_.end()) {
107  LOG(ERROR) << "Invalid argument, stream=" << GetStreamLabel(pair.first)
108  << " not available.";
109  return Status(error::INVALID_ARGUMENT, "Stream not available");
110  }
111  }
112 
113  while (!cancelled_ && status.ok())
114  status.Update(Parse());
115  if (cancelled_ && status.ok())
116  return Status(error::CANCELLED, "Demuxer run cancelled");
117 
118  LOG(ERROR) << "status error code" << status.error_code();
119  if (status.error_code() == error::END_OF_STREAM) {
120  for (size_t stream_index : stream_indexes_) {
121  status = FlushDownstream(stream_index);
122  if (!status.ok())
123  return status;
124  }
125  return Status::OK;
126  }
127  return status;
128 }

152 Status Demuxer::InitializeParser() {
153  DCHECK(!media_file_);
154  DCHECK(!all_streams_ready_);
155 
156  LOG(INFO) << "Initialize Demuxer for file '" << file_name_ << "'.";
157 
158  media_file_ = File::Open(file_name_.c_str(), "r");
159  if (!media_file_) {
160  return Status(error::FILE_FAILURE,
161  "Cannot open file for reading " + file_name_);
162  }
163 
164  // Read enough bytes before detecting the container.
165  int64_t bytes_read = 0;
166  while (static_cast<size_t>(bytes_read) < kInitBufSize) {
167  int64_t read_result =
168  media_file_->Read(buffer_.get() + bytes_read, kInitBufSize);
169  if (read_result < 0)
170  return Status(error::FILE_FAILURE, "Cannot read file " + file_name_);
171  if (read_result == 0)
172  break;
173  bytes_read += read_result;
174  }
175  container_name_ = DetermineContainer(buffer_.get(), bytes_read);
176 
177  // Initialize media parser.
178  switch (container_name_) {
179  case CONTAINER_MOV:
180  parser_.reset(new mp4::MP4MediaParser());
181  break;
182  case CONTAINER_MPEG2TS:
183  parser_.reset(new mp2t::Mp2tMediaParser());
184  break;
185  // Widevine classic (WVM) is derived from MPEG2PS. We do not support
186  // non-WVM MPEG2PS file, thus we do not differentiate between the two.
187  // Every MPEG2PS file is assumed to be WVM file. If it turns out not the
188  // case, an error will be reported when trying to parse the file as WVM
189  // file.
190  case CONTAINER_MPEG2PS:
191  FALLTHROUGH_INTENDED;
192  case CONTAINER_WVM:
193  parser_.reset(new wvm::WvmMediaParser());
194  break;
195  case CONTAINER_WEBM:
196  parser_.reset(new WebMMediaParser());
197  break;
198  case CONTAINER_WEBVTT:
199  parser_.reset(new WebVttParser());
200  break;
201  case CONTAINER_UNKNOWN: {
202  const int64_t kDumpSizeLimit = 512;
203  LOG(ERROR) << "Failed to detect the container type from the buffer: "
204  << base::HexEncode(buffer_.get(),
205  std::min(bytes_read, kDumpSizeLimit));
206  return Status(error::INVALID_ARGUMENT,
207  "Failed to detect the container type.");
208  }
209  default:
210  NOTIMPLEMENTED() << "Container " << container_name_
211  << " is not supported.";
212  return Status(error::UNIMPLEMENTED, "Container not supported.");
213  }
214 
215  parser_->Init(
216  base::Bind(&Demuxer::ParserInitEvent, base::Unretained(this)),
217  base::Bind(&Demuxer::NewMediaSampleEvent, base::Unretained(this)),
218  base::Bind(&Demuxer::NewTextSampleEvent, base::Unretained(this)),
219  key_source_.get());
220 
221  // Handle trailing 'moov'.
222  if (container_name_ == CONTAINER_MOV &&
223  File::IsLocalRegularFile(file_name_.c_str())) {
224  // TODO(kqyang): Investigate whether we can reuse the existing file
225  // descriptor |media_file_| instead of opening the same file again.
226  static_cast<mp4::MP4MediaParser*>(parser_.get())->LoadMoov(file_name_);
227  }
228  if (!parser_->Parse(buffer_.get(), bytes_read)) {
229  return Status(error::PARSER_FAILURE,
230  "Cannot parse media file " + file_name_);
231  }
232  return Status::OK;
233 }

235 void Demuxer::ParserInitEvent(
236  const std::vector<std::shared_ptr<StreamInfo>>& stream_infos) {
237  if (dump_stream_info_) {
238  printf("\nFile \"%s\":\n", file_name_.c_str());
239  printf("Found %zu stream(s).\n", stream_infos.size());
240  for (size_t i = 0; i < stream_infos.size(); ++i)
241  printf("Stream [%zu] %s\n", i, stream_infos[i]->ToString().c_str());
242  }
243 
244  int base_stream_index = 0;
245  bool video_handler_set =
246  output_handlers().find(kBaseVideoOutputStreamIndex) !=
247  output_handlers().end();
248  bool audio_handler_set =
249  output_handlers().find(kBaseAudioOutputStreamIndex) !=
250  output_handlers().end();
251  bool text_handler_set =
252  output_handlers().find(kBaseTextOutputStreamIndex) !=
253  output_handlers().end();
254  for (const std::shared_ptr<StreamInfo>& stream_info : stream_infos) {
255  size_t stream_index = base_stream_index;
256  if (video_handler_set && stream_info->stream_type() == kStreamVideo) {
257  stream_index = kBaseVideoOutputStreamIndex;
258  // Only for the first video stream.
259  video_handler_set = false;
260  }
261  if (audio_handler_set && stream_info->stream_type() == kStreamAudio) {
262  stream_index = kBaseAudioOutputStreamIndex;
263  // Only for the first audio stream.
264  audio_handler_set = false;
265  }
266  if (text_handler_set && stream_info->stream_type() == kStreamText) {
267  stream_index = kBaseTextOutputStreamIndex;
268  text_handler_set = false;
269  }
270 
271  const bool handler_set =
272  output_handlers().find(stream_index) != output_handlers().end();
273  if (handler_set) {
274  track_id_to_stream_index_map_[stream_info->track_id()] = stream_index;
275  stream_indexes_.push_back(stream_index);
276  auto iter = language_overrides_.find(stream_index);
277  if (iter != language_overrides_.end() &&
278  stream_info->stream_type() != kStreamVideo) {
279  stream_info->set_language(iter->second);
280  }
281  if (stream_info->is_encrypted()) {
282  init_event_status_.Update(Status(error::INVALID_ARGUMENT,
283  "A decryption key source is not "
284  "provided for an encrypted stream."));
285  } else {
286  init_event_status_.Update(
287  DispatchStreamInfo(stream_index, stream_info));
288  }
289  } else {
290  track_id_to_stream_index_map_[stream_info->track_id()] =
291  kInvalidStreamIndex;
292  }
293  ++base_stream_index;
294  }
295  all_streams_ready_ = true;
296 }

298 bool Demuxer::NewMediaSampleEvent(uint32_t track_id,
299  std::shared_ptr<MediaSample> sample) {
300  if (!all_streams_ready_) {
301  if (queued_media_samples_.size() >= kQueuedSamplesLimit) {
302  LOG(ERROR) << "Queued samples limit reached: " << kQueuedSamplesLimit;
303  return false;
304  }
305  queued_media_samples_.emplace_back(track_id, sample);
306  return true;
307  }
308  if (!init_event_status_.ok()) {
309  return false;
310  }
311 
312  while (!queued_media_samples_.empty()) {
313  if (!PushMediaSample(queued_media_samples_.front().track_id,
314  queued_media_samples_.front().sample)) {
315  return false;
316  }
317  queued_media_samples_.pop_front();
318  }
319  return PushMediaSample(track_id, sample);
320 }

346 bool Demuxer::PushMediaSample(uint32_t track_id,
347  std::shared_ptr<MediaSample> sample) {
348  auto stream_index_iter = track_id_to_stream_index_map_.find(track_id);
349  if (stream_index_iter == track_id_to_stream_index_map_.end()) {
350  LOG(ERROR) << "Track " << track_id << " not found.";
351  return false;
352  }
353  if (stream_index_iter->second == kInvalidStreamIndex)
354  return true;
355  Status status = DispatchMediaSample(stream_index_iter->second, sample);
356  if (!status.ok()) {
357  LOG(ERROR) << "Failed to process sample " << stream_index_iter->second
358  << " " << status;
359  return false;
360  }
361  return true;
362 }

382 Status Demuxer::Parse() {
383  DCHECK(media_file_);
384  DCHECK(parser_);
385  DCHECK(buffer_);
386 
387  int64_t bytes_read = media_file_->Read(buffer_.get(), kBufSize);
388  if (bytes_read == 0) {
389  if (!parser_->Flush())
390  return Status(error::PARSER_FAILURE, "Failed to flush.");
391  return Status(error::END_OF_STREAM, "");
392  } else if (bytes_read < 0) {
393  return Status(error::FILE_FAILURE, "Cannot read file " + file_name_);
394  }
395 
396  return parser_->Parse(buffer_.get(), bytes_read)
397  ? Status::OK
398  : Status(error::PARSER_FAILURE,
399  "Cannot parse media file " + file_name_);
400 }
