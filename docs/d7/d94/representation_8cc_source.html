24 std::string GetMimeType(const std::string& prefix,
25  MediaInfo::ContainerType container_type) {
26  switch (container_type) {
27  case MediaInfo::CONTAINER_MP4:
28  return prefix + "/mp4";
29  case MediaInfo::CONTAINER_MPEG2_TS:
30  // NOTE: DASH MPD spec uses lowercase but RFC3555 says uppercase.
31  return prefix + "/MP2T";
32  case MediaInfo::CONTAINER_WEBM:
33  return prefix + "/webm";
34  default:
35  break;
36  }
37 
38  // Unsupported container types should be rejected/handled by the caller.
39  LOG(ERROR) << "Unrecognized container type: " << container_type;
40  return std::string();
41 }

45 // width, height, framerate, and sar.

47  if (!video_info.has_height() || !video_info.has_width()) {
48  LOG(ERROR)

57  LOG_IF(WARNING, !video_info.has_pixel_width())
58  << "Video info does not contain pixel_width to calculate the sample "
59  "aspect ratio required for DASH IOP.";
60  LOG_IF(WARNING, !video_info.has_pixel_height())
61  << "Video info does not contain pixel_height to calculate the sample "
62  "aspect ratio required for DASH IOP.";

117 bool Representation::Init() {
118  if (!AtLeastOneTrue(media_info_.has_video_info(),
119  media_info_.has_audio_info(),
120  media_info_.has_text_info())) {
121  // This is an error. Segment information can be in AdaptationSet, Period, or
122  // MPD but the interface does not provide a way to set them.
123  // See 5.3.9.1 ISO 23009-1:2012 for segment info.
124  LOG(ERROR) << "Representation needs one of video, audio, or text.";
125  return false;
126  }
127 
128  if (MoreThanOneTrue(media_info_.has_video_info(),
129  media_info_.has_audio_info(),
130  media_info_.has_text_info())) {
131  LOG(ERROR) << "Only one of VideoInfo, AudioInfo, or TextInfo can be set.";
132  return false;
133  }
134 
135  if (media_info_.container_type() == MediaInfo::CONTAINER_UNKNOWN) {
136  LOG(ERROR) << "'container_type' in MediaInfo cannot be CONTAINER_UNKNOWN.";
137  return false;
138  }
139 
140  if (media_info_.has_video_info()) {
141  mime_type_ = GetVideoMimeType();
142  if (!HasRequiredVideoFields(media_info_.video_info())) {
143  LOG(ERROR) << "Missing required fields to create a video Representation.";
144  return false;
145  }
146  } else if (media_info_.has_audio_info()) {
147  mime_type_ = GetAudioMimeType();
148  } else if (media_info_.has_text_info()) {
149  mime_type_ = GetTextMimeType();
150  }
151 
152  if (mime_type_.empty())
153  return false;
154 
155  codecs_ = GetCodecs(media_info_);
156  return true;
157 }

171 void Representation::AddNewSegment(int64_t start_time,
172  int64_t duration,
173  uint64_t size) {
174  if (start_time == 0 && duration == 0) {
175  LOG(WARNING) << "Got segment with start_time and duration == 0. Ignoring.";
176  return;
177  }
178 
179  // In order for the oldest segment to be accessible for at least
180  // |time_shift_buffer_depth| seconds, the latest segment should not be in the
181  // sliding window since the player could be playing any part of the latest
182  // segment. So the current segment duration is added to the sum of segment
183  // durations (in the manifest/playlist) after sliding the window.
184  SlideWindow();
185 
186  if (state_change_listener_)
187  state_change_listener_->OnNewSegmentForRepresentation(start_time, duration);
188 
189  AddSegmentInfo(start_time, duration);
190 
191  // Only update the buffer depth and bandwidth estimator when the full segment
192  // is completed. In the low latency case, only the first chunk in the segment
193  // has been written at this point. Therefore, we must wait until the entire
194  // segment has been written before updating buffer depth and bandwidth
195  // estimator.
196  if (!mpd_options_.mpd_params.low_latency_dash_mode) {
197  current_buffer_depth_ += segment_infos_.back().duration;
198 
199  bandwidth_estimator_.AddBlock(size, static_cast<double>(duration) /
200  media_info_.reference_time_scale());
201  }
202 }

246 // Uses info in |media_info_| and |content_protection_elements_| to create a
247 // "Representation" node.
248 // MPD schema has strict ordering. The following must be done in order.
249 // AddVideoInfo() (possibly adds FramePacking elements), AddAudioInfo() (Adds
250 // AudioChannelConfig elements), AddContentProtectionElements*(), and
251 // AddVODOnlyInfo() (Adds segment info).
252 base::Optional<xml::XmlNode> Representation::GetXml() {
253  if (!HasRequiredMediaInfoFields()) {
254  LOG(ERROR) << "MediaInfo missing required fields.";
255  return base::nullopt;
256  }
257 
258  const uint64_t bandwidth = media_info_.has_bandwidth()
259  ? media_info_.bandwidth()
260  : bandwidth_estimator_.Max();
261 
262  DCHECK(!(HasVODOnlyFields(media_info_) && HasLiveOnlyFields(media_info_)));
263 
264  xml::RepresentationXmlNode representation;
265  // Mandatory fields for Representation.
266  if (!representation.SetId(id_) ||
267  !representation.SetIntegerAttribute("bandwidth", bandwidth) ||
268  !(codecs_.empty() ||
269  representation.SetStringAttribute("codecs", codecs_)) ||
270  !representation.SetStringAttribute("mimeType", mime_type_)) {
271  return base::nullopt;
272  }
273 
274  const bool has_video_info = media_info_.has_video_info();
275  const bool has_audio_info = media_info_.has_audio_info();
276 
277  if (has_video_info &&
278  !representation.AddVideoInfo(
279  media_info_.video_info(),
280  !(output_suppression_flags_ & kSuppressWidth),
281  !(output_suppression_flags_ & kSuppressHeight),
282  !(output_suppression_flags_ & kSuppressFrameRate))) {
283  LOG(ERROR) << "Failed to add video info to Representation XML.";
284  return base::nullopt;
285  }
286 
287  if (has_audio_info &&
288  !representation.AddAudioInfo(media_info_.audio_info())) {
289  LOG(ERROR) << "Failed to add audio info to Representation XML.";
290  return base::nullopt;
291  }
292 
293  if (!representation.AddContentProtectionElements(
294  content_protection_elements_)) {
295  return base::nullopt;
296  }
297 
298  if (HasVODOnlyFields(media_info_) &&
299  !representation.AddVODOnlyInfo(
300  media_info_, mpd_options_.mpd_params.use_segment_list,
301  mpd_options_.mpd_params.target_segment_duration)) {
302  LOG(ERROR) << "Failed to add VOD info.";
303  return base::nullopt;
304  }
305 
306  if (HasLiveOnlyFields(media_info_) &&
307  !representation.AddLiveOnlyInfo(
308  media_info_, segment_infos_, start_number_,
309  mpd_options_.mpd_params.low_latency_dash_mode)) {
310  LOG(ERROR) << "Failed to add Live info.";
311  return base::nullopt;
312  }
313  // TODO(rkuroiwa): It is likely that all representations have the exact same
314  // SegmentTemplate. Optimize and propagate the tag up to AdaptationSet level.
315 
316  output_suppression_flags_ = 0;
317  return std::move(representation);
318 }

332 void Representation::SetAvailabilityTimeOffset() {
333  // Adjust the frame duration to units of seconds to match target segment
334  // duration.
335  const double frame_duration_sec =
336  (double)frame_duration_ / (double)media_info_.reference_time_scale();
337  // availabilityTimeOffset = segment duration - chunk duration.
338  // Here, the frame duration is equivalent to the sample duration,
339  // see Representation::SetSampleDuration(uint32_t frame_duration).
340  // By definition, each chunk will contain only one sample;
341  // thus, chunk_duration = sample_duration = frame_duration.
342  const double ato =
343  mpd_options_.mpd_params.target_segment_duration - frame_duration_sec;
344  if (ato <= 0)
345  return;
346  media_info_.set_availability_time_offset(ato);
347 }

384 void Representation::AddSegmentInfo(int64_t start_time, int64_t duration) {
385  const uint64_t kNoRepeat = 0;
386  const int64_t adjusted_duration = AdjustDuration(duration);
387 
388  if (!segment_infos_.empty()) {
389  // Contiguous segment.
390  const SegmentInfo& previous = segment_infos_.back();
391  const int64_t previous_segment_end_time =
392  previous.start_time + previous.duration * (previous.repeat + 1);
393  // Make it continuous if the segment start time is close to previous segment
394  // end time.
395  if (ApproximiatelyEqual(previous_segment_end_time, start_time)) {
396  const int64_t segment_end_time_for_same_duration =
397  previous_segment_end_time + previous.duration;
398  const int64_t actual_segment_end_time = start_time + duration;
399  // Consider the segments having identical duration if the segment end time
400  // is close to calculated segment end time by assuming identical duration.
401  if (ApproximiatelyEqual(segment_end_time_for_same_duration,
402  actual_segment_end_time)) {
403  ++segment_infos_.back().repeat;
404  } else {
405  segment_infos_.push_back(
406  {previous_segment_end_time,
407  actual_segment_end_time - previous_segment_end_time, kNoRepeat});
408  }
409  return;
410  }
411 
412  // A gap since previous.
413  const int64_t kRoundingErrorGrace = 5;
414  if (previous_segment_end_time + kRoundingErrorGrace < start_time) {
415  LOG(WARNING) << RepresentationAsString() << " Found a gap of size "
416  << (start_time - previous_segment_end_time)
417  << " > kRoundingErrorGrace (" << kRoundingErrorGrace
418  << "). The new segment starts at " << start_time
419  << " but the previous segment ends at "
420  << previous_segment_end_time << ".";
421  }
422 
423  // No overlapping segments.
424  if (start_time < previous_segment_end_time - kRoundingErrorGrace) {
425  LOG(WARNING)
426  << RepresentationAsString()
427  << " Segments should not be overlapping. The new segment starts at "
428  << start_time << " but the previous segment ends at "
429  << previous_segment_end_time << ".";
430  }
431  }
432 
433  segment_infos_.push_back({start_time, adjusted_duration, kNoRepeat});
434 }

443 bool Representation::ApproximiatelyEqual(int64_t time1, int64_t time2) const {
444  if (!allow_approximate_segment_timeline_)
445  return time1 == time2;
446 
447  // It is not always possible to align segment duration to target duration
448  // exactly. For example, for AAC with sampling rate of 44100, there are always
449  // 1024 audio samples per frame, so the frame duration is 1024/44100. For a
450  // target duration of 2 seconds, the closest segment duration would be 1.984
451  // or 2.00533.
452 
453  // An arbitrary error threshold cap. This makes sure that the error is not too
454  // large for large samples.
455  const double kErrorThresholdSeconds = 0.05;
456 
457  // So we consider two times equal if they differ by less than one sample.
458  const int32_t error_threshold =
459  std::min(frame_duration_,
460  static_cast<int32_t>(kErrorThresholdSeconds *
461  media_info_.reference_time_scale()));
462  return std::abs(time1 - time2) <= error_threshold;
463 }

465 int64_t Representation::AdjustDuration(int64_t duration) const {

476 void Representation::SlideWindow() {
477  if (mpd_options_.mpd_params.time_shift_buffer_depth <= 0.0 ||
478  mpd_options_.mpd_type == MpdType::kStatic)
479  return;
480 
481  const int32_t time_scale = GetTimeScale(media_info_);
482  DCHECK_GT(time_scale, 0);
483 
484  const int64_t time_shift_buffer_depth = static_cast<int64_t>(
485  mpd_options_.mpd_params.time_shift_buffer_depth * time_scale);
486 
487  if (current_buffer_depth_ <= time_shift_buffer_depth)
488  return;
489 
490  std::list<SegmentInfo>::iterator first = segment_infos_.begin();
491  std::list<SegmentInfo>::iterator last = first;
492  for (; last != segment_infos_.end(); ++last) {
493  // Remove the current segment only if it falls completely out of time shift
494  // buffer range.
495  while (last->repeat >= 0 &&
496  current_buffer_depth_ - last->duration >= time_shift_buffer_depth) {
497  current_buffer_depth_ -= last->duration;
498  RemoveOldSegment(&*last);
499  start_number_++;
500  }
501  if (last->repeat >= 0)
502  break;
503  }
504  segment_infos_.erase(first, last);
505 }

507 void Representation::RemoveOldSegment(SegmentInfo* segment_info) {
508  int64_t segment_start_time = segment_info->start_time;
509  segment_info->start_time += segment_info->duration;
510  segment_info->repeat--;
511 
512  if (mpd_options_.mpd_params.preserved_segments_outside_live_window == 0)
513  return;
514 
515  segments_to_be_removed_.push_back(
516  media::GetSegmentName(media_info_.segment_template(), segment_start_time,
517  start_number_ - 1, media_info_.bandwidth()));
518  while (segments_to_be_removed_.size() >
519  mpd_options_.mpd_params.preserved_segments_outside_live_window) {
520  VLOG(2) << "Deleting " << segments_to_be_removed_.front();
521  if (!File::Delete(segments_to_be_removed_.front().c_str())) {
522  LOG(WARNING) << "Failed to delete " << segments_to_be_removed_.front()
523  << "; Will retry later.";
524  break;
525  }
526  segments_to_be_removed_.pop_front();
527  }
