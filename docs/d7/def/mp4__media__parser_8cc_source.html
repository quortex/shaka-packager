140 bool UpdateCodecStringForDolbyVision(
141  FourCC actual_format,
142  const std::vector<CodecConfiguration>& configs,
143  std::string* codec_string) {
144  DOVIDecoderConfigurationRecord dovi_config;
145  if (!dovi_config.Parse(GetDOVIDecoderConfig(configs))) {
146  LOG(ERROR) << "Failed to parse Dolby Vision decoder "
147  "configuration record.";
148  return false;
149  }
150  switch (actual_format) {
151  case FOURCC_dvh1:
152  case FOURCC_dvhe:
153  // Non-Backward compatibility mode. Replace the code string with
154  // Dolby Vision only.
155  *codec_string = dovi_config.GetCodecString(actual_format);
156  break;
157  case FOURCC_hev1:
158  // Backward compatibility mode. Two codecs are signalled: base codec
159  // without Dolby Vision and HDR with Dolby Vision.
160  *codec_string += ";" + dovi_config.GetCodecString(FOURCC_dvhe);
161  break;
162  case FOURCC_hvc1:
163  // See above.
164  *codec_string += ";" + dovi_config.GetCodecString(FOURCC_dvh1);
165  break;
166  default:
167  LOG(ERROR) << "Unsupported format with extra codec "
168  << FourCCToString(actual_format);
169  return false;
170  }
171  return true;
172 }

217 bool MP4MediaParser::Parse(const uint8_t* buf, int size) {
218  DCHECK_NE(state_, kWaitingForInit);
219 
220  if (state_ == kError)
221  return false;
222 
223  queue_.Push(buf, size);
224 
225  bool result, err = false;
226 
227  do {
228  if (state_ == kParsingBoxes) {
229  result = ParseBox(&err);
230  } else {
231  DCHECK_EQ(kEmittingSamples, state_);
232  result = EnqueueSample(&err);
233  if (result) {
234  int64_t max_clear = runs_->GetMaxClearOffset() + moof_head_;
235  err = !ReadAndDiscardMDATsUntil(max_clear);
236  }
237  }
238  } while (result && !err);
239 
240  if (err) {
241  DLOG(ERROR) << "Error while parsing MP4";
242  moov_.reset();
243  Reset();
244  ChangeState(kError);
245  return false;
246  }
247 
248  return true;
249 }

251 bool MP4MediaParser::LoadMoov(const std::string& file_path) {
252  std::unique_ptr<File, FileCloser> file(
253  File::OpenWithNoBuffering(file_path.c_str(), "r"));
254  if (!file) {
255  LOG(ERROR) << "Unable to open media file '" << file_path << "'";
256  return false;
257  }
258  if (!file->Seek(0)) {
259  LOG(WARNING) << "Filesystem does not support seeking on file '" << file_path
260  << "'";
261  return false;
262  }
263 
264  uint64_t file_position(0);
265  bool mdat_seen(false);
266  while (true) {
267  const uint32_t kBoxHeaderReadSize(16);
268  std::vector<uint8_t> buffer(kBoxHeaderReadSize);
269  int64_t bytes_read = file->Read(&buffer[0], kBoxHeaderReadSize);
270  if (bytes_read == 0) {
271  LOG(ERROR) << "Could not find 'moov' box in file '" << file_path << "'";
272  return false;
273  }
274  if (bytes_read < kBoxHeaderReadSize) {
275  LOG(ERROR) << "Error reading media file '" << file_path << "'";
276  return false;
277  }
278  uint64_t box_size;
279  FourCC box_type;
280  bool err;
281  if (!BoxReader::StartBox(&buffer[0], kBoxHeaderReadSize, &box_type,
282  &box_size, &err)) {
283  LOG(ERROR) << "Could not start box from file '" << file_path << "'";
284  return false;
285  }
286  if (box_type == FOURCC_mdat) {
287  mdat_seen = true;
288  } else if (box_type == FOURCC_moov) {
289  if (!mdat_seen) {
290  // 'moov' is before 'mdat'. Nothing to do.
291  break;
292  }
293  // 'mdat' before 'moov'. Read and parse 'moov'.
294  if (!Parse(&buffer[0], bytes_read)) {
295  LOG(ERROR) << "Error parsing mp4 file '" << file_path << "'";
296  return false;
297  }
298  uint64_t bytes_to_read = box_size - bytes_read;
299  buffer.resize(bytes_to_read);
300  while (bytes_to_read > 0) {
301  bytes_read = file->Read(&buffer[0], bytes_to_read);
302  if (bytes_read <= 0) {
303  LOG(ERROR) << "Error reading 'moov' contents from file '" << file_path
304  << "'";
305  return false;
306  }
307  if (!Parse(&buffer[0], bytes_read)) {
308  LOG(ERROR) << "Error parsing mp4 file '" << file_path << "'";
309  return false;
310  }
311  bytes_to_read -= bytes_read;
312  }
313  queue_.Reset(); // So that we don't need to adjust data offsets.
314  mdat_tail_ = 0; // So it will skip boxes until mdat.
315  break; // Done.
316  }
317  file_position += box_size;
318  if (!file->Seek(file_position)) {
319  LOG(ERROR) << "Error skipping box in mp4 file '" << file_path << "'";
320  return false;
321  }
322  }
323  return true;
324 }

326 bool MP4MediaParser::ParseBox(bool* err) {
327  const uint8_t* buf;
328  int size;
329  queue_.Peek(&buf, &size);
330  if (!size)
331  return false;
332 
333  std::unique_ptr<BoxReader> reader(BoxReader::ReadBox(buf, size, err));
334  if (reader.get() == NULL)
335  return false;
336 
337  if (reader->type() == FOURCC_mdat) {
338  if (!moov_) {
339  // For seekable files, we seek to the 'moov' and load the 'moov' first
340  // then seek back (see LoadMoov function for details); we do not support
341  // having 'mdat' before 'moov' for non-seekable files. The code ends up
342  // here only if it is a non-seekable file.
343  NOTIMPLEMENTED() << " Non-seekable Files with 'mdat' box before 'moov' "
344  "box is not supported.";
345  *err = true;
346  return false;
347  } else {
348  // This can happen if there are unused 'mdat' boxes, which is unusual
349  // but allowed by the spec. Ignore the 'mdat' and proceed.
350  LOG(INFO)
351  << "Ignore unused 'mdat' box - this could be as a result of extra "
352  "not usable 'mdat' or 'mdat' associated with unrecognized track.";
353  }
354  }
355 
356  // Set up mdat offset for ReadMDATsUntil().
357  mdat_tail_ = queue_.head() + reader->size();
358 
359  if (reader->type() == FOURCC_moov) {
360  *err = !ParseMoov(reader.get());
361  } else if (reader->type() == FOURCC_moof) {
362  moof_head_ = queue_.head();
363  *err = !ParseMoof(reader.get());
364 
365  // Return early to avoid evicting 'moof' data from queue. Auxiliary info may
366  // be located anywhere in the file, including inside the 'moof' itself.
367  // (Since 'default-base-is-moof' is mandated, no data references can come
368  // before the head of the 'moof', so keeping this box around is sufficient.)
369  return !(*err);
370  } else {
371  VLOG(2) << "Skipping top-level box: " << FourCCToString(reader->type());
372  }
373 
374  queue_.Pop(static_cast<int>(reader->size()));
375  return !(*err);
376 }

378 bool MP4MediaParser::ParseMoov(BoxReader* reader) {
379  if (moov_)
380  return true; // Already parsed the 'moov' box.
381 
382  moov_.reset(new Movie);
383  RCHECK(moov_->Parse(reader));
384  runs_.reset();
385 
386  std::vector<std::shared_ptr<StreamInfo>> streams;
387 
388  for (std::vector<Track>::const_iterator track = moov_->tracks.begin();
389  track != moov_->tracks.end(); ++track) {
390  const int32_t timescale = track->media.header.timescale;
391 
392  // Calculate duration (based on timescale).
393  int64_t duration = 0;
394  if (track->media.header.duration > 0) {
395  duration = track->media.header.duration;
396  } else if (moov_->extends.header.fragment_duration > 0) {
397  DCHECK(moov_->header.timescale != 0);
398  duration = Rescale(moov_->extends.header.fragment_duration,
399  moov_->header.timescale,
400  timescale);
401  } else if (moov_->header.duration > 0 &&
402  moov_->header.duration != std::numeric_limits<uint64_t>::max()) {
403  DCHECK(moov_->header.timescale != 0);
404  duration =
405  Rescale(moov_->header.duration, moov_->header.timescale, timescale);
406  }
407 
408  const SampleDescription& samp_descr =
409  track->media.information.sample_table.description;
410 
411  size_t desc_idx = 0;
412 
413  // Read sample description index from mvex if it exists otherwise read
414  // from the first entry in Sample To Chunk box.
415  if (moov_->extends.tracks.size() > 0) {
416  for (size_t t = 0; t < moov_->extends.tracks.size(); t++) {
417  const TrackExtends& trex = moov_->extends.tracks[t];
418  if (trex.track_id == track->header.track_id) {
419  desc_idx = trex.default_sample_description_index;
420  break;
421  }
422  }
423  } else {
424  const std::vector<ChunkInfo>& chunk_info =
425  track->media.information.sample_table.sample_to_chunk.chunk_info;
426  RCHECK(chunk_info.size() > 0);
427  desc_idx = chunk_info[0].sample_description_index;
428  }
429  RCHECK(desc_idx > 0);
430  desc_idx -= 1; // BMFF descriptor index is one-based
431 
432  if (samp_descr.type == kAudio) {
433  RCHECK(!samp_descr.audio_entries.empty());
434 
435  // It is not uncommon to find otherwise-valid files with incorrect sample
436  // description indices, so we fail gracefully in that case.
437  if (desc_idx >= samp_descr.audio_entries.size())
438  desc_idx = 0;
439 
440  const AudioSampleEntry& entry = samp_descr.audio_entries[desc_idx];
441  const FourCC actual_format = entry.GetActualFormat();
442  Codec codec = FourCCToCodec(actual_format);
443  uint8_t num_channels = entry.channelcount;
444  uint32_t sampling_frequency = entry.samplerate;
445  uint64_t codec_delay_ns = 0;
446  uint8_t audio_object_type = 0;
447  uint32_t max_bitrate = 0;
448  uint32_t avg_bitrate = 0;
449  std::vector<uint8_t> codec_config;
450 
451  switch (actual_format) {
452  case FOURCC_mp4a: {
453  const DecoderConfigDescriptor& decoder_config =
454  entry.esds.es_descriptor.decoder_config_descriptor();
455  max_bitrate = decoder_config.max_bitrate();
456  avg_bitrate = decoder_config.avg_bitrate();
457 
458  codec = ObjectTypeToCodec(decoder_config.object_type());
459  if (codec == kCodecAAC) {
460  const AACAudioSpecificConfig& aac_audio_specific_config =
461  entry.esds.aac_audio_specific_config;
462  num_channels = aac_audio_specific_config.GetNumChannels();
463  sampling_frequency =
464  aac_audio_specific_config.GetSamplesPerSecond();
465  audio_object_type = aac_audio_specific_config.GetAudioObjectType();
466  codec_config =
467  decoder_config.decoder_specific_info_descriptor().data();
468  } else if (codec == kUnknownCodec) {
469  // Intentionally not to fail in the parser as there may be multiple
470  // streams in the source content, which allows the supported stream
471  // to be packaged. An error will be returned if the unsupported
472  // stream is passed to the muxer.
473  LOG(WARNING) << "Unsupported audio object type "
474  << static_cast<int>(decoder_config.object_type())
475  << " in stsd.es_desriptor.";
476  }
477  break;
478  }
479  case FOURCC_dtsc:
480  FALLTHROUGH_INTENDED;
481  case FOURCC_dtse:
482  FALLTHROUGH_INTENDED;
483  case FOURCC_dtsh:
484  FALLTHROUGH_INTENDED;
485  case FOURCC_dtsl:
486  FALLTHROUGH_INTENDED;
487  case FOURCC_dtsm:
488  codec_config = entry.ddts.extra_data;
489  max_bitrate = entry.ddts.max_bitrate;
490  avg_bitrate = entry.ddts.avg_bitrate;
491  break;
492  case FOURCC_ac_3:
493  codec_config = entry.dac3.data;
494  num_channels = static_cast<uint8_t>(GetAc3NumChannels(codec_config));
495  break;
496  case FOURCC_ec_3:
497  codec_config = entry.dec3.data;
498  num_channels = static_cast<uint8_t>(GetEc3NumChannels(codec_config));
499  break;
500  case FOURCC_ac_4:
501  codec_config = entry.dac4.data;
502  // Stop the process if have errors when parsing AC-4 dac4 box,
503  // bitstream version 0 (has beed deprecated) and contains multiple
504  // presentations in single AC-4 stream (only used for broadcast).
505  if (!GetAc4CodecInfo(codec_config, &audio_object_type)) {
506  LOG(ERROR) << "Failed to parse dac4.";
507  return false;
508  }
509  break;
510  case FOURCC_fLaC:
511  codec_config = entry.dfla.data;
512  break;
513  case FOURCC_Opus:
514  codec_config = entry.dops.opus_identification_header;
515  codec_delay_ns =
516  entry.dops.preskip * kNanosecondsPerSecond / sampling_frequency;
517  break;
518  case FOURCC_mha1:
519  case FOURCC_mhm1:
520  codec_config = entry.mhac.data;
521  audio_object_type = entry.mhac.mpeg_h_3da_profile_level_indication;
522  break;
523  default:
524  // Intentionally not to fail in the parser as there may be multiple
525  // streams in the source content, which allows the supported stream to
526  // be packaged.
527  // An error will be returned if the unsupported stream is passed to
528  // the muxer.
529  LOG(WARNING) << "Unsupported audio format '"
530  << FourCCToString(actual_format) << "' in stsd box.";
531  break;
532  }
533 
534  // Extract possible seek preroll.
535  uint64_t seek_preroll_ns = 0;
536  for (const auto& sample_group_description :
537  track->media.information.sample_table.sample_group_descriptions) {
538  if (sample_group_description.grouping_type != FOURCC_roll)
539  continue;
540  const auto& audio_roll_recovery_entries =
541  sample_group_description.audio_roll_recovery_entries;
542  if (audio_roll_recovery_entries.size() != 1) {
543  LOG(WARNING) << "Unexpected number of entries in "
544  "SampleGroupDescription table with grouping type "
545  "'roll'.";
546  break;
547  }
548  const int16_t roll_distance_in_samples =
549  audio_roll_recovery_entries[0].roll_distance;
550  if (roll_distance_in_samples < 0) {
551  RCHECK(sampling_frequency != 0);
552  seek_preroll_ns = kNanosecondsPerSecond *
553  (-roll_distance_in_samples) / sampling_frequency;
554  } else {
555  LOG(WARNING)
556  << "Roll distance is supposed to be negative, but seeing "
557  << roll_distance_in_samples;
558  }
559  break;
560  }
561 
562  // The stream will be decrypted if a |decryptor_source_| is available.
563  const bool is_encrypted =
564  decryptor_source_
565  ? false
566  : entry.sinf.info.track_encryption.default_is_protected == 1;
567  DVLOG(1) << "is_audio_track_encrypted_: " << is_encrypted;
568  streams.emplace_back(new AudioStreamInfo(
569  track->header.track_id, timescale, duration, codec,
570  AudioStreamInfo::GetCodecString(codec, audio_object_type),
571  codec_config.data(), codec_config.size(), entry.samplesize,
572  num_channels, sampling_frequency, seek_preroll_ns, codec_delay_ns,
573  max_bitrate, avg_bitrate, track->media.header.language.code,
574  is_encrypted));
575  }
576 
577  if (samp_descr.type == kVideo) {
578  RCHECK(!samp_descr.video_entries.empty());
579  if (desc_idx >= samp_descr.video_entries.size())
580  desc_idx = 0;
581  const VideoSampleEntry& entry = samp_descr.video_entries[desc_idx];
582  std::vector<uint8_t> codec_configuration_data =
583  entry.codec_configuration.data;
584 
585  uint32_t coded_width = entry.width;
586  uint32_t coded_height = entry.height;
587  uint32_t pixel_width = entry.pixel_aspect.h_spacing;
588  uint32_t pixel_height = entry.pixel_aspect.v_spacing;
589  if (pixel_width == 0 && pixel_height == 0) {
590  DerivePixelWidthHeight(coded_width, coded_height, track->header.width,
591  track->header.height, &pixel_width,
592  &pixel_height);
593  }
594  std::string codec_string;
595  uint8_t nalu_length_size = 0;
596  uint8_t transfer_characteristics = 0;
597 
598  const FourCC actual_format = entry.GetActualFormat();
599  const Codec video_codec = FourCCToCodec(actual_format);
600  switch (actual_format) {
601  case FOURCC_av01: {
602  AV1CodecConfigurationRecord av1_config;
603  if (!av1_config.Parse(codec_configuration_data)) {
604  LOG(ERROR) << "Failed to parse av1c.";
605  return false;
606  }
607  codec_string = av1_config.GetCodecString();
608  break;
609  }
610  case FOURCC_avc1:
611  case FOURCC_avc3: {
612  AVCDecoderConfigurationRecord avc_config;
613  if (!avc_config.Parse(codec_configuration_data)) {

783 bool MP4MediaParser::EnqueueSample(bool* err) {
784  if (!runs_->IsRunValid()) {
785  // Remain in kEnqueueingSamples state, discarding data, until the end of
786  // the current 'mdat' box has been appended to the queue.
787  if (!queue_.Trim(mdat_tail_))
788  return false;
789 
790  ChangeState(kParsingBoxes);
791  return true;
792  }
793 
794  if (!runs_->IsSampleValid()) {
795  runs_->AdvanceRun();
796  return true;
797  }
798 
799  DCHECK(!(*err));
800 
801  const uint8_t* buf;
802  int buf_size;
803  queue_.Peek(&buf, &buf_size);
804  if (!buf_size)
805  return false;
806 
807  // Skip this entire track if it is not audio nor video.
808  if (!runs_->is_audio() && !runs_->is_video())
809  runs_->AdvanceRun();
810 
811  // Attempt to cache the auxiliary information first. Aux info is usually
812  // placed in a contiguous block before the sample data, rather than being
813  // interleaved. If we didn't cache it, this would require that we retain the
814  // start of the segment buffer while reading samples. Aux info is typically
815  // quite small compared to sample data, so this pattern is useful on
816  // memory-constrained devices where the source buffer consumes a substantial
817  // portion of the total system memory.
818  if (runs_->AuxInfoNeedsToBeCached()) {
819  queue_.PeekAt(runs_->aux_info_offset() + moof_head_, &buf, &buf_size);
820  if (buf_size < runs_->aux_info_size())
821  return false;
822  *err = !runs_->CacheAuxInfo(buf, buf_size);
823  return !*err;
824  }
825 
826  int64_t sample_offset = runs_->sample_offset() + moof_head_;
827  queue_.PeekAt(sample_offset, &buf, &buf_size);
828  if (buf_size < runs_->sample_size()) {
829  if (sample_offset < queue_.head()) {
830  LOG(ERROR) << "Incorrect sample offset " << sample_offset
831  << " < " << queue_.head();
832  *err = true;
833  }
834  return false;
835  }
836 
837  const uint8_t* media_data = buf;
838  const size_t media_data_size = runs_->sample_size();
839  // Use a dummy data size of 0 to avoid copying overhead.
840  // Actual media data is set later.
841  const size_t kDummyDataSize = 0;
842  std::shared_ptr<MediaSample> stream_sample(
843  MediaSample::CopyFrom(media_data, kDummyDataSize, runs_->is_keyframe()));
844 
845  if (runs_->is_encrypted()) {
846  std::shared_ptr<uint8_t> decrypted_media_data(
847  new uint8_t[media_data_size], std::default_delete<uint8_t[]>());
848  std::unique_ptr<DecryptConfig> decrypt_config = runs_->GetDecryptConfig();
849  if (!decrypt_config) {
850  *err = true;
851  LOG(ERROR) << "Missing decrypt config.";
852  return false;
853  }
854 
855  if (!decryptor_source_) {
856  stream_sample->SetData(media_data, media_data_size);
857  // If the demuxer does not have the decryptor_source_, store
858  // decrypt_config so that the demuxed sample can be decrypted later.
859  stream_sample->set_decrypt_config(std::move(decrypt_config));
860  stream_sample->set_is_encrypted(true);
861  } else {
862  if (!decryptor_source_->DecryptSampleBuffer(decrypt_config.get(),
863  media_data, media_data_size,
864  decrypted_media_data.get())) {
865  *err = true;
866  LOG(ERROR) << "Cannot decrypt samples.";
867  return false;
868  }
869  stream_sample->TransferData(std::move(decrypted_media_data),
870  media_data_size);
871  }
872  } else {
873  stream_sample->SetData(media_data, media_data_size);
874  }
875 
876  stream_sample->set_dts(runs_->dts());
877  stream_sample->set_pts(runs_->cts());
878  stream_sample->set_duration(runs_->duration());
879 
880  DVLOG(3) << "Pushing frame: "
881  << ", key=" << runs_->is_keyframe()
882  << ", dur=" << runs_->duration()
883  << ", dts=" << runs_->dts()
884  << ", cts=" << runs_->cts()
885  << ", size=" << runs_->sample_size();
886 
887  if (!new_sample_cb_.Run(runs_->track_id(), stream_sample)) {
888  *err = true;
889  LOG(ERROR) << "Failed to process the sample.";
890  return false;
891  }
892 
893  runs_->AdvanceSample();
894  return true;
895 }

897 bool MP4MediaParser::ReadAndDiscardMDATsUntil(const int64_t offset) {
898  bool err = false;
899  while (mdat_tail_ < offset) {
900  const uint8_t* buf;
901  int size;
902  queue_.PeekAt(mdat_tail_, &buf, &size);
903 
904  FourCC type;
905  uint64_t box_sz;
906  if (!BoxReader::StartBox(buf, size, &type, &box_sz, &err))
907  break;
908 
909  mdat_tail_ += box_sz;
910  }
911  queue_.Trim(std::min(mdat_tail_, offset));
912  return !err;
913 }
