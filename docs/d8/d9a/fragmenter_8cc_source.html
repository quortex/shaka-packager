65 Status Fragmenter::AddSample(const MediaSample& sample) {
66  const int64_t pts = sample.pts();
67  const int64_t dts = sample.dts();
68  const int64_t duration = sample.duration();
69  if (duration == 0)
70  LOG(WARNING) << "Unexpected sample with zero duration @ dts " << dts;
71 
72  if (!fragment_initialized_)
73  RETURN_IF_ERROR(InitializeFragment(dts));
74 
75  if (sample.side_data_size() > 0)
76  LOG(WARNING) << "MP4 samples do not support side data. Side data ignored.";
77 
78  // Fill in sample parameters. It will be optimized later.
79  traf_->runs[0].sample_sizes.push_back(
80  static_cast<uint32_t>(sample.data_size()));
81  traf_->runs[0].sample_durations.push_back(duration);
82  traf_->runs[0].sample_flags.push_back(
83  sample.is_key_frame() ? 0 : TrackFragmentHeader::kNonKeySampleMask);
84 
85  if (sample.decrypt_config()) {
86  NewSampleEncryptionEntry(
87  *sample.decrypt_config(),
88  !stream_info_->encryption_config().constant_iv.empty(), traf_);
89  }
90 
91  if (stream_info_->stream_type() == StreamType::kStreamVideo &&
92  sample.is_key_frame()) {
93  key_frame_infos_.push_back({pts, data_->Size(), sample.data_size()});
94  }
95 
96  data_->AppendArray(sample.data(), sample.data_size());
97 
98  traf_->runs[0].sample_composition_time_offsets.push_back(pts - dts);
99  if (pts != dts)
100  traf_->runs[0].flags |= TrackFragmentRun::kSampleCompTimeOffsetsPresentMask;
101 
102  // Exclude the part of sample with negative pts out of duration calculation as
103  // they are not presented.
104  if (pts < 0) {
105  const int64_t end_pts = pts + duration;
106  if (end_pts > 0) {
107  // Include effective presentation duration.
108  fragment_duration_ += end_pts;
109 
110  earliest_presentation_time_ = 0;
111  if (sample.is_key_frame())
112  first_sap_time_ = 0;
113  }
114  } else {
115  fragment_duration_ += duration;
116 
117  if (earliest_presentation_time_ > pts)
118  earliest_presentation_time_ = pts;
119 
120  if (sample.is_key_frame()) {
121  if (first_sap_time_ == kInvalidTime)
122  first_sap_time_ = pts;
123  }
124  }
125  return Status::OK;
126 }

128 Status Fragmenter::InitializeFragment(int64_t first_sample_dts) {
129  fragment_initialized_ = true;
130  fragment_finalized_ = false;
131 
132  // |first_sample_dts| is adjusted by the edit list offset. The offset should
133  // be un-applied in |decode_time|, so when applying the Edit List, the result
134  // dts is |first_sample_dts|.
135  const int64_t dts_before_edit = first_sample_dts + edit_list_offset_;
136  traf_->decode_time.decode_time = dts_before_edit;
137 
138  traf_->runs.clear();
139  traf_->runs.resize(1);
140  traf_->runs[0].flags = TrackFragmentRun::kDataOffsetPresentMask;
141  traf_->auxiliary_size.sample_info_sizes.clear();
142  traf_->auxiliary_offset.offsets.clear();
143  traf_->sample_encryption.sample_encryption_entries.clear();
144  traf_->sample_group_descriptions.clear();
145  traf_->sample_to_groups.clear();
146  traf_->header.sample_description_index = 1; // 1-based.
147  traf_->header.flags = TrackFragmentHeader::kDefaultBaseIsMoofMask |
148  TrackFragmentHeader::kSampleDescriptionIndexPresentMask;
149 
150  fragment_duration_ = 0;
151  earliest_presentation_time_ = kInvalidTime;
152  first_sap_time_ = kInvalidTime;
153  data_.reset(new BufferWriter());
154  key_frame_infos_.clear();
155  return Status::OK;
156 }

158 Status Fragmenter::FinalizeFragment() {
159  if (!fragment_initialized_)
160  return Status::OK;
161 
162  if (stream_info_->is_encrypted()) {
163  Status status = FinalizeFragmentForEncryption();
164  if (!status.ok())
165  return status;
166  }
167 
168  // Optimize trun box.
169  traf_->runs[0].sample_count =
170  static_cast<uint32_t>(traf_->runs[0].sample_sizes.size());
171  if (OptimizeSampleEntries(&traf_->runs[0].sample_durations,
172  &traf_->header.default_sample_duration)) {
173  traf_->header.flags |=
174  TrackFragmentHeader::kDefaultSampleDurationPresentMask;
175  } else {
176  traf_->runs[0].flags |= TrackFragmentRun::kSampleDurationPresentMask;
177  }
178  if (OptimizeSampleEntries(&traf_->runs[0].sample_sizes,
179  &traf_->header.default_sample_size)) {
180  traf_->header.flags |= TrackFragmentHeader::kDefaultSampleSizePresentMask;
181  } else {
182  traf_->runs[0].flags |= TrackFragmentRun::kSampleSizePresentMask;
183  }
184  if (OptimizeSampleEntries(&traf_->runs[0].sample_flags,
185  &traf_->header.default_sample_flags)) {
186  traf_->header.flags |= TrackFragmentHeader::kDefaultSampleFlagsPresentMask;
187  } else {
188  traf_->runs[0].flags |= TrackFragmentRun::kSampleFlagsPresentMask;
189  }
190 
191  // Add SampleToGroup boxes. A SampleToGroup box with grouping type of 'roll'
192  // needs to be added if there is seek preroll, referencing sample group
193  // description in track level; Also need to add SampleToGroup boxes
194  // correponding to every SampleGroupDescription boxes, referencing sample
195  // group description in fragment level.
196  DCHECK_EQ(traf_->sample_to_groups.size(), 0u);
197  if (seek_preroll_ > 0) {
198  traf_->sample_to_groups.resize(traf_->sample_to_groups.size() + 1);
199  SampleToGroup& sample_to_group = traf_->sample_to_groups.back();
200  sample_to_group.grouping_type = FOURCC_roll;
201 
202  sample_to_group.entries.resize(1);
203  SampleToGroupEntry& sample_to_group_entry = sample_to_group.entries.back();
204  sample_to_group_entry.sample_count = traf_->runs[0].sample_count;
205  sample_to_group_entry.group_description_index =
206  SampleToGroupEntry::kTrackGroupDescriptionIndexBase + 1;
207  }
208  for (const auto& sample_group_description :
209  traf_->sample_group_descriptions) {
210  traf_->sample_to_groups.resize(traf_->sample_to_groups.size() + 1);
211  SampleToGroup& sample_to_group = traf_->sample_to_groups.back();
212  sample_to_group.grouping_type = sample_group_description.grouping_type;
213 
214  sample_to_group.entries.resize(1);
215  SampleToGroupEntry& sample_to_group_entry = sample_to_group.entries.back();
216  sample_to_group_entry.sample_count = traf_->runs[0].sample_count;
217  sample_to_group_entry.group_description_index =
218  SampleToGroupEntry::kTrackFragmentGroupDescriptionIndexBase + 1;
219  }
220 
221  fragment_finalized_ = true;
222  fragment_initialized_ = false;
223  return Status::OK;
224 }

226 void Fragmenter::GenerateSegmentReference(SegmentReference* reference) const {
227  // NOTE: Daisy chain is not supported currently.
228  reference->reference_type = false;
229  reference->subsegment_duration = fragment_duration_;
230  reference->starts_with_sap = StartsWithSAP();
231  if (kInvalidTime == first_sap_time_) {

233  reference->sap_delta_time = 0;

241 Status Fragmenter::FinalizeFragmentForEncryption() {
242  SampleEncryption& sample_encryption = traf_->sample_encryption;
243  if (sample_encryption.sample_encryption_entries.empty()) {
244  // This fragment is not encrypted.
245  // There are two sample description entries, an encrypted entry and a clear
246  // entry, are generated. The 1-based clear entry index is always 2.
247  const uint32_t kClearSampleDescriptionIndex = 2;
248  traf_->header.sample_description_index = kClearSampleDescriptionIndex;
249  return Status::OK;
250  }
251  if (sample_encryption.sample_encryption_entries.size() !=
252  traf_->runs[0].sample_sizes.size()) {
253  LOG(ERROR) << "Partially encrypted segment is not supported";
254  return Status(error::MUXER_FAILURE,
255  "Partially encrypted segment is not supported.");
256  }
257 
258  const SampleEncryptionEntry& sample_encryption_entry =
259  sample_encryption.sample_encryption_entries.front();
260  const bool use_subsample_encryption =
261  !sample_encryption_entry.subsamples.empty();
262  if (use_subsample_encryption)
263  traf_->sample_encryption.flags |= SampleEncryption::kUseSubsampleEncryption;
264  traf_->sample_encryption.iv_size = static_cast<uint8_t>(
265  sample_encryption_entry.initialization_vector.size());
266 
267  // The offset will be adjusted in Segmenter after knowing moof size.
268  traf_->auxiliary_offset.offsets.push_back(0);
269 
270  // Optimize saiz box.
271  SampleAuxiliaryInformationSize& saiz = traf_->auxiliary_size;
272  saiz.sample_count = static_cast<uint32_t>(saiz.sample_info_sizes.size());
273  DCHECK_EQ(saiz.sample_info_sizes.size(),
274  traf_->sample_encryption.sample_encryption_entries.size());
275  if (!OptimizeSampleEntries(&saiz.sample_info_sizes,
276  &saiz.default_sample_info_size)) {
277  saiz.default_sample_info_size = 0;
278  }
279 
280  // It should only happen with full sample encryption + constant iv, i.e.
281  // 'cbcs' applying to audio.
282  if (saiz.default_sample_info_size == 0 && saiz.sample_info_sizes.empty()) {
283  DCHECK(!use_subsample_encryption);
284  // ISO/IEC 23001-7:2016(E) The sample auxiliary information would then be
285  // empty and should be omitted. Clear saiz and saio boxes so they are not
286  // written.
287  saiz.sample_count = 0;
288  traf_->auxiliary_offset.offsets.clear();
289  }
290  return Status::OK;
291 }

293 bool Fragmenter::StartsWithSAP() const {
294  DCHECK(!traf_->runs.empty());
295  uint32_t start_sample_flag;
296  if (traf_->runs[0].flags & TrackFragmentRun::kSampleFlagsPresentMask) {
297  DCHECK(!traf_->runs[0].sample_flags.empty());
298  start_sample_flag = traf_->runs[0].sample_flags[0];
299  } else {
300  DCHECK(traf_->header.flags &
301  TrackFragmentHeader::kDefaultSampleFlagsPresentMask);
302  start_sample_flag = traf_->header.default_sample_flags;
303  }
304  return (start_sample_flag & TrackFragmentHeader::kNonKeySampleMask) == 0;
305 }
