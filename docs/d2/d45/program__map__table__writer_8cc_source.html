105 // Note there are dozens of CRCs. This is one of them.
106 // http://reveng.sourceforge.net/crc-catalogue/all.htm
107 uint32_t Crc32Mpeg2(const uint8_t* data, size_t data_size) {
108  uint32_t crc = 0xFFFFFFFF;
109  for (size_t i = 0; i < data_size; ++i) {
110  crc = kCrcTable[((crc >> 24) ^ data[i]) & 0xFF] ^ (crc << 8);
111  }
112  return crc;
113 }

134 bool WriteRegistrationDescriptorForEncryptedAudio(Codec codec,
135  const uint8_t* setup_data,
136  size_t setup_data_size,
137  BufferWriter* output) {
138  const uint8_t kRegistrationDescriptor = 5;
139  BufferWriter audio_setup_information;
140  if (!WriteAudioSetupInformation(codec, setup_data, setup_data_size,
141  &audio_setup_information)) {
142  return false;
143  }
144 
145  const size_t registration_descriptor_size =
146  audio_setup_information.Size() + sizeof(FOURCC_apad);
147  if (registration_descriptor_size > std::numeric_limits<uint8_t>::max()) {
148  LOG(ERROR) << "Audio setup data of size: " << setup_data_size
149  << " will not fit in the descriptor.";
150  return false;
151  }
152 
153  output->AppendInt(kRegistrationDescriptor);
154  output->AppendInt(static_cast<uint8_t>(registration_descriptor_size));
155  output->AppendInt(FOURCC_apad);
156  output->AppendBuffer(audio_setup_information);
157  return true;
158 }

160 void WritePmtWithParameters(uint8_t stream_type,
161  int version,
162  int current_next_indicator,
163  const uint8_t* descriptors,
164  size_t descriptors_size,
165  BufferWriter* pmt) {
166  DCHECK(current_next_indicator == kCurrent || current_next_indicator == kNext);
167  // Body starting from program number.
168  BufferWriter pmt_body;
169  pmt_body.AppendInt(static_cast<uint8_t>(0x00));
170  pmt_body.AppendInt(kProgramNumber);
171  // resevered bits then version and current_next_indicator.
172  pmt_body.AppendInt(
173  static_cast<uint8_t>(0xC0 |
174  static_cast<uint8_t>(version) << 1 |
175  static_cast<uint8_t>(current_next_indicator)));
176  // section number.
177  pmt_body.AppendInt(static_cast<uint8_t>(0x00));
178  // last section number.
179  pmt_body.AppendInt(static_cast<uint8_t>(0x00));
180  // first 3 bits reserved. Rest is unused bits for PCR PID.
181  pmt_body.AppendInt(static_cast<uint8_t>(0xE0));
182  pmt_body.AppendInt(ProgramMapTableWriter::kElementaryPid);
183  // First 4 bits are reserved. Next 12 bits is program_info_length which is 0.
184  pmt_body.AppendInt(static_cast<uint8_t>(0xF0));
185  pmt_body.AppendInt(static_cast<uint8_t>(0x00));
186 
187  pmt_body.AppendInt(stream_type);
188  // 3 reserved bits followed by 13 bit elementary_PID.
189  pmt_body.AppendInt(static_cast<uint8_t>(0xE0));
190  pmt_body.AppendInt(ProgramMapTableWriter::kElementaryPid);
191 
192  // 4 reserved bits followed by ES_info_length.
193  pmt_body.AppendInt(static_cast<uint16_t>(0xF000 | descriptors_size));
194  if (descriptors_size > 0) {
195  DCHECK(descriptors);
196  pmt_body.AppendArray(descriptors, descriptors_size);
197  }
198 
199  pmt->Clear();
200  // Pointer field is not really part of the PMT but it's there so that an extra
201  // buffer isn't required to prepend the 0x00 byte.
202  const uint8_t kPointerField = 0;
203  pmt->AppendInt(kPointerField);
204  pmt->AppendInt(kProgramMapTableId);
205  // First four bits must be '1011'. +4 for CRC.
206  pmt->AppendInt(static_cast<uint16_t>(0xB000 | (pmt_body.Size() + 4)));
207  pmt->AppendBuffer(pmt_body);
208 
209  // Don't include the pointer field.
210  const uint32_t crc = Crc32Mpeg2(pmt->Buffer() + 1, pmt->Size() - 1);
211  pmt->AppendInt(crc);
212 }

218 bool ProgramMapTableWriter::EncryptedSegmentPmt(BufferWriter* writer) {
219  if (encrypted_pmt_.Size() == 0) {
220  TsStreamType stream_type;
221  switch (codec_) {
222  case kCodecH264:
223  stream_type = TsStreamType::kEncryptedAvc;
224  break;
225  case kCodecAAC:
226  stream_type = TsStreamType::kEncryptedAdtsAac;
227  break;
228  case kCodecAC3:
229  stream_type = TsStreamType::kEncryptedAc3;
230  break;
231  case kCodecEAC3:
232  stream_type = TsStreamType::kEncryptedEac3;
233  break;
234  default:
235  LOG(ERROR) << "Codec " << codec_ << " is not supported in TS yet.";
236  return false;
237  }
238 
239  BufferWriter descriptors;
240  if (!WriteDescriptors(&descriptors))
241  return false;
242 
243  const bool has_clear_lead = clear_pmt_.Size() > 0;
244  WritePmtWithParameters(static_cast<uint8_t>(stream_type),
245  has_clear_lead ? kVersion1 : kVersion0, kCurrent,
246  descriptors.Buffer(), descriptors.Size(),
247  &encrypted_pmt_);
248  DCHECK_NE(encrypted_pmt_.Size(), 0u);
249  }
250  WritePmtToBuffer(encrypted_pmt_.Buffer(), encrypted_pmt_.Size(),
251  &continuity_counter_, writer);
252  return true;
253 }

314 bool AudioProgramMapTableWriter::WriteDescriptors(
315  BufferWriter* descriptors) const {
316  FourCC fourcc;
317  switch (codec()) {
318  case kCodecAAC:
319  fourcc = FOURCC_aacd;
320  break;
321  case kCodecMP3:
322  fourcc = FOURCC_mp3a;
323  break;
324  case kCodecAC3:
325  fourcc = FOURCC_ac3d;
326  break;
327  case kCodecEAC3:
328  fourcc = FOURCC_ec3d;
329  break;
330  default:
331  LOG(ERROR) << "Codec " << codec() << " is not supported in TS yet.";
332  return false;
333  }
334  WritePrivateDataIndicatorDescriptor(fourcc, descriptors);
335 
336  // NOTE: There are two specifications of carrying AC-3 bit stream in MPEG-2
337  // transport stream (ISO/IEC 13818-1):
338  // System A used by ATSC (TS 102 366 Digital Audio Compression Standard)
339  // stream_type: 0x81
340  // system_id: 0xBD (private_stream_1)
341  // Requires Registration_descriptor, AC-3_audio_stream_descriptor.
342  // Optional ISO_639_language_code descriptor.
343  // System B used by DVB (TS 101 154 DVB specification for ... based on the
344  // MPEG-2 Transport Stream)
345  // stream_type: 0x06 (private data)
346  // stream_id: 0xBD (private_stream_1)
347  // Requires AC-3_descriptor (not the same as AC-3_audio_stream_descriptor
348  // in ATSC)
349  // Optional ISO_639_language_code descriptor.
350  // We follow "System A" but not strictly as we do not include Registration
351  // descriptor and AC-3_audio_stream_descriptor right now.
352 
353  return WriteRegistrationDescriptorForEncryptedAudio(
354  codec(), audio_specific_config_.data(), audio_specific_config_.size(),
355  descriptors);
356 }
