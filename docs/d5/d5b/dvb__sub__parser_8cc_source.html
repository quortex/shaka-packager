19 RgbaColor ConvertYuv(uint8_t Y, uint8_t Cr, uint8_t Cb, uint8_t T) {
20  // Converts based on ITU-R BT.601.
21  // See https://en.wikipedia.org/wiki/YCbCr
22  //
23  // Note that the T value should be interpolated based on a full transparency
24  // being 256. This means that T=255 should not be fully transparent. Y=0 is
25  // used to signal full transparency.
26  // Values for Y<16 (except Y=0) are invalid, so clamp to 16.
27  RgbaColor color;
28  const double y_transform = 255.0 / 219 * (std::max<uint8_t>(Y, 16) - 16);
29  const double cb_transform = 255.0 / 244 * 1.772 * (Cb - 128);
30  const double cr_transform = 255.0 / 244 * 1.402 * (Cr - 128);
31  const double f1 = 0.114 / 0.587;
32  const double f2 = 0.299 / 0.587;
33  color.r = static_cast<uint8_t>(y_transform + cr_transform);
34  color.g =
35  static_cast<uint8_t>(y_transform - cb_transform * f1 - cr_transform * f2);
36  color.b = static_cast<uint8_t>(y_transform + cb_transform);
37  color.a = Y == 0 ? 0 : (T == 0 ? 255 : 256 - T);
38  return color;
39 }

48 bool DvbSubParser::Parse(DvbSubSegmentType segment_type,
49  int64_t pts,
50  const uint8_t* payload,
51  size_t size,
52  std::vector<std::shared_ptr<TextSample>>* samples) {
53  switch (segment_type) {
54  case DvbSubSegmentType::kPageComposition:
55  return ParsePageComposition(pts, payload, size, samples);
56  case DvbSubSegmentType::kRegionComposition:
57  return ParseRegionComposition(payload, size);
58  case DvbSubSegmentType::kClutDefinition:
59  return ParseClutDefinition(payload, size);
60  case DvbSubSegmentType::kObjectData:
61  return ParseObjectData(pts, payload, size);
62  case DvbSubSegmentType::kDisplayDefinition:
63  return ParseDisplayDefinition(payload, size);
64  case DvbSubSegmentType::kEndOfDisplay:
65  // This signals all the current objects are available. But we need to
66  // know the end time, so we do nothing for now.
67  return true;
68  default:
69  LOG(WARNING) << "Unknown DVB-sub segment_type=0x" << std::hex
70  << static_cast<uint32_t>(segment_type);
71  return true;
72  }
73 }

92 bool DvbSubParser::ParsePageComposition(
93  int64_t pts,
94  const uint8_t* data,
95  size_t size,
96  std::vector<std::shared_ptr<TextSample>>* samples) {
97  // See ETSI EN 300 743 Section 7.2.2.
98  BitReader reader(data, size);
99 
100  uint8_t page_state;
101  RCHECK(reader.ReadBits(8, &timeout_));
102  RCHECK(reader.SkipBits(4)); // page_version_number
103  RCHECK(reader.ReadBits(2, &page_state));
104  RCHECK(reader.SkipBits(2)); // reserved
105  if (page_state == 0x1 || page_state == 0x2) {
106  int64_t end_pts_ = std::min(last_pts_ + previous_timeout_ * kMpeg2Timescale, pts);
107  LOG(INFO) << "clean object : min(pts_end, timeout) : " << end_pts_ << " pts_end : " << pts;
108  RCHECK(composer_.GetSamples(last_pts_, end_pts_, samples));
109  composer_.ClearObjects();
110  last_pts_ = pts;
111  previous_timeout_ = timeout_;
112  }
113 
114  while (reader.bits_available() > 0u) {
115  uint8_t region_id;
116  uint16_t x, y;
117  RCHECK(reader.ReadBits(8, &region_id));
118  RCHECK(reader.SkipBits(8)); // reserved
119  RCHECK(reader.ReadBits(16, &x));
120  RCHECK(reader.ReadBits(16, &y));
121 
122  RCHECK(composer_.SetRegionPosition(region_id, x, y));
123  }
124 
125  return true;
126 }

128 bool DvbSubParser::ParseRegionComposition(const uint8_t* data, size_t size) {
129  // See ETSI EN 300 743 Section 7.2.3.
130  BitReader reader(data, size);
131 
132  uint8_t region_id, clut_id;
133  uint16_t region_width, region_height;
134  bool region_fill_flag;
135  int background_pixel_code;
136  RCHECK(reader.ReadBits(8, &region_id));
137  RCHECK(reader.SkipBits(4)); // region_version_number
138  RCHECK(reader.ReadBits(1, &region_fill_flag));
139  RCHECK(reader.SkipBits(3)); // reserved
140  RCHECK(reader.ReadBits(16, &region_width));
141  RCHECK(reader.ReadBits(16, &region_height));
142  RCHECK(reader.SkipBits(3)); // region_level_of_compatibility
143  RCHECK(reader.SkipBits(3)); // region_depth
144  RCHECK(reader.SkipBits(2)); // reserved
145  RCHECK(reader.ReadBits(8, &clut_id));
146  RCHECK(reader.ReadBits(8, &background_pixel_code));
147  RCHECK(reader.SkipBits(4)); // region_4-bit_pixel_code
148  RCHECK(reader.SkipBits(2)); // region_2-bit_pixel_code
149  RCHECK(reader.SkipBits(2)); // reserved
150  RCHECK(
151  composer_.SetRegionInfo(region_id, clut_id, region_width, region_height));
152  if (!region_fill_flag)
153  background_pixel_code = -1;
154 
155  while (reader.bits_available() > 0) {
156  uint16_t object_id, x, y;
157  uint8_t object_type;
158  RCHECK(reader.ReadBits(16, &object_id));
159  RCHECK(reader.ReadBits(2, &object_type));
160  RCHECK(reader.SkipBits(2)); // object_provider_flag
161  RCHECK(reader.ReadBits(12, &x));
162  RCHECK(reader.SkipBits(4)); // reserved
163  RCHECK(reader.ReadBits(12, &y));
164 
165  if (object_type == 0x01 || object_type == 0x02) {
166  RCHECK(reader.SkipBits(8)); // foreground_pixel_code
167  RCHECK(reader.SkipBits(8)); // background_pixel_code
168  }
169  RCHECK(composer_.SetObjectInfo(object_id, region_id, x, y,
170  background_pixel_code));
171  }
172 
173  return true;
174 }

176 bool DvbSubParser::ParseClutDefinition(const uint8_t* data, size_t size) {
177  // See ETSI EN 300 743 Section 7.2.4.
178  BitReader reader(data, size);
179 
180  uint8_t clut_id;
181  RCHECK(reader.ReadBits(8, &clut_id));
182  auto* color_space = composer_.GetColorSpace(clut_id);
183  RCHECK(reader.SkipBits(4)); // CLUT_version_number
184  RCHECK(reader.SkipBits(4)); // reserved
185  while (reader.bits_available() > 0) {
186  uint8_t clut_entry_id;
187  uint8_t has_2_bit;
188  uint8_t has_4_bit;
189  uint8_t has_8_bit;
190  uint8_t full_range_flag;
191  RCHECK(reader.ReadBits(8, &clut_entry_id));
192  RCHECK(reader.ReadBits(1, &has_2_bit));
193  RCHECK(reader.ReadBits(1, &has_4_bit));
194  RCHECK(reader.ReadBits(1, &has_8_bit));
195  RCHECK(reader.SkipBits(4)); // reserved
196  RCHECK(reader.ReadBits(1, &full_range_flag));
197 
198  if (has_2_bit + has_4_bit + has_8_bit != 1) {
199  LOG(ERROR) << "Must specify exactly one bit depth in CLUT definition";
200  return false;
201  }
202  const BitDepth bit_depth =
203  has_2_bit ? BitDepth::k2Bit
204  : (has_4_bit ? BitDepth::k4Bit : BitDepth::k8Bit);
205 
206  uint8_t Y, Cr, Cb, T;
207  if (full_range_flag) {
208  RCHECK(reader.ReadBits(8, &Y));
209  RCHECK(reader.ReadBits(8, &Cr));
210  RCHECK(reader.ReadBits(8, &Cb));
211  RCHECK(reader.ReadBits(8, &T));
212  } else {
213  // These store the most-significant bits, so shift them up.
214  RCHECK(reader.ReadBits(6, &Y));
215  Y <<= 2;
216  RCHECK(reader.ReadBits(4, &Cr));
217  Cr <<= 4;
218  RCHECK(reader.ReadBits(4, &Cb));
219  Cb <<= 4;
220  RCHECK(reader.ReadBits(2, &T));
221  T <<= 6;
222  }
223  color_space->SetColor(bit_depth, clut_entry_id, ConvertYuv(Y, Cr, Cb, T));
224  }
225 
226  return true;
227 }

229 bool DvbSubParser::ParseObjectData(int64_t pts,
230  const uint8_t* data,
231  size_t size) {
232  // See ETSI EN 300 743 Section 7.2.5 Table 17.
233  BitReader reader(data, size);
234 
235  uint16_t object_id;
236  uint8_t object_coding_method;
237  RCHECK(reader.ReadBits(16, &object_id));
238  RCHECK(reader.SkipBits(4)); // object_version_number
239  RCHECK(reader.ReadBits(2, &object_coding_method));
240  RCHECK(reader.SkipBits(1)); // non_modifying_colour_flag
241  RCHECK(reader.SkipBits(1)); // reserved
242 
243  auto* image = composer_.GetObjectImage(object_id);
244  auto* color_space = composer_.GetColorSpaceForObject(object_id);
245  if (!image || !color_space)
246  return false;
247 
248  if (object_coding_method == 0) {
249  uint16_t top_field_length;
250  uint16_t bottom_field_length;
251  RCHECK(reader.ReadBits(16, &top_field_length));
252  RCHECK(reader.ReadBits(16, &bottom_field_length));
253 
254  RCHECK(ParsePixelDataSubObject(top_field_length, true, &reader, color_space,
255  image));
256  RCHECK(ParsePixelDataSubObject(bottom_field_length, false, &reader,
257  color_space, image));
258  // Ignore 8_stuff_bits since we don't need to read to the end.
259 
260  if (bottom_field_length == 0) {
261  // If there are no bottom rows, then the top rows are used instead. See
262  // beginning of section 7.2.5.1.
263  image->MirrorToBottomRows();
264  }
265  } else {
266  LOG(ERROR) << "Unsupported DVB-sub object coding method: "
267  << static_cast<int>(object_coding_method);
268  return false;
269  }
270  return true;
271 }

274  // See ETSI EN 300 743 Section 7.2.1.
275  BitReader reader(data, size);
276 
277  uint16_t width, height;
278  RCHECK(reader.SkipBits(4)); // dds_version_number

289 bool DvbSubParser::ParsePixelDataSubObject(size_t sub_object_length,
290  bool is_top_fields,
291  BitReader* reader,
292  DvbImageColorSpace* color_space,
293  DvbImageBuilder* image) {
294  const size_t start = reader->bit_position() / 8;
295  while (reader->bit_position() / 8 < start + sub_object_length) {
296  // See ETSI EN 300 743 Section 7.2.5.1 Table 20
297  uint8_t data_type;
298  RCHECK(reader->ReadBits(8, &data_type));
299  uint8_t temp[16];
300  switch (data_type) {
301  case 0x10:
302  RCHECK(Parse2BitPixelData(is_top_fields, reader, image));
303  reader->SkipToNextByte();
304  break;
305  case 0x11:
306  RCHECK(Parse4BitPixelData(is_top_fields, reader, image));
307  reader->SkipToNextByte();
308  break;
309  case 0x12:
310  RCHECK(Parse8BitPixelData(is_top_fields, reader, image));
311  break;
312  case 0x20:
313  for (int i = 0; i < 4; i++) {
314  RCHECK(reader->ReadBits(4, &temp[i]));
315  }
316  color_space->Set2To4BitDepthMap(temp);
317  break;
318  case 0x21:
319  for (int i = 0; i < 4; i++) {
320  RCHECK(reader->ReadBits(8, &temp[i]));
321  }
322  color_space->Set2To8BitDepthMap(temp);
323  break;
324  case 0x22:
325  for (int i = 0; i < 16; i++) {
326  RCHECK(reader->ReadBits(8, &temp[i]));
327  }
328  color_space->Set4To8BitDepthMap(temp);
329  break;
330  case 0xf0:
331  image->NewRow(is_top_fields);
332  break;
333  default:
334  LOG(ERROR) << "Unsupported DVB-sub pixel data format: 0x" << std::hex
335  << static_cast<int>(data_type);
336  return false;
337  }
338  }
339  return true;
340 }

342 bool DvbSubParser::Parse2BitPixelData(bool is_top_fields,
343  BitReader* reader,
344  DvbImageBuilder* image) {
345  // 2-bit/pixel code string, Section 7.2.5.2.1, Table 22.
346  while (true) {
347  uint8_t peek;
348  RCHECK(reader->ReadBits(2, &peek));
349  if (peek != 0) {
350  RCHECK(image->AddPixel(BitDepth::k2Bit, peek, is_top_fields));
351  } else {
352  uint8_t switch_1;
353  RCHECK(reader->ReadBits(1, &switch_1));
354  if (switch_1 == 1) {
355  uint8_t count_minus_3;
356  RCHECK(reader->ReadBits(3, &count_minus_3));
357  RCHECK(reader->ReadBits(2, &peek));
358  for (uint8_t i = 0; i < count_minus_3 + 3; i++)
359  RCHECK(image->AddPixel(BitDepth::k2Bit, peek, is_top_fields));
360  } else {
361  uint8_t switch_2;
362  RCHECK(reader->ReadBits(1, &switch_2));
363  if (switch_2 == 1) {
364  RCHECK(image->AddPixel(BitDepth::k2Bit, 0, is_top_fields));
365  } else {
366  uint8_t switch_3;
367  RCHECK(reader->ReadBits(2, &switch_3));
368  if (switch_3 == 0) {
369  break;
370  } else if (switch_3 == 1) {
371  RCHECK(image->AddPixel(BitDepth::k2Bit, 0, is_top_fields));
372  RCHECK(image->AddPixel(BitDepth::k2Bit, 0, is_top_fields));
373  } else if (switch_3 == 2) {
374  uint8_t count_minus_12;
375  RCHECK(reader->ReadBits(4, &count_minus_12));
376  RCHECK(reader->ReadBits(2, &peek));
377  for (uint8_t i = 0; i < count_minus_12 + 12; i++)
378  RCHECK(image->AddPixel(BitDepth::k2Bit, peek, is_top_fields));
379  } else if (switch_3 == 3) {
380  uint8_t count_minus_29;
381  RCHECK(reader->ReadBits(8, &count_minus_29));
382  RCHECK(reader->ReadBits(2, &peek));
383  for (uint8_t i = 0; i < count_minus_29 + 29; i++)
384  RCHECK(image->AddPixel(BitDepth::k2Bit, peek, is_top_fields));
385  }
386  }
387  }
388  }
389  }
390 
391  return true;
392 }

394 bool DvbSubParser::Parse4BitPixelData(bool is_top_fields,
395  BitReader* reader,
396  DvbImageBuilder* image) {
397  // 4-bit/pixel code string, Section 7.2.5.2.2, Table 24.
398  DCHECK(reader->bits_available() % 8 == 0);
399  while (true) {
400  uint8_t peek;
401  RCHECK(reader->ReadBits(4, &peek));
402  if (peek != 0) {
403  RCHECK(image->AddPixel(BitDepth::k4Bit, peek, is_top_fields));
404  } else {
405  uint8_t switch_1;
406  RCHECK(reader->ReadBits(1, &switch_1));
407  if (switch_1 == 0) {
408  RCHECK(reader->ReadBits(3, &peek));
409  if (peek != 0) {
410  for (int i = 0; i < peek + 2; i++)
411  RCHECK(image->AddPixel(BitDepth::k4Bit, 0, is_top_fields));
412  } else {
413  break;
414  }
415  } else {
416  uint8_t switch_2;
417  RCHECK(reader->ReadBits(1, &switch_2));
418  if (switch_2 == 0) {
419  RCHECK(reader->ReadBits(2, &peek)); // run_length_4-7
420  uint8_t code;
421  RCHECK(reader->ReadBits(4, &code));
422  for (int i = 0; i < peek + 4; i++)
423  RCHECK(image->AddPixel(BitDepth::k4Bit, code, is_top_fields));
424  } else {
425  uint8_t switch_3;
426  RCHECK(reader->ReadBits(2, &switch_3));
427  if (switch_3 == 0) {
428  RCHECK(image->AddPixel(BitDepth::k4Bit, 0, is_top_fields));
429  } else if (switch_3 == 1) {
430  RCHECK(image->AddPixel(BitDepth::k4Bit, 0, is_top_fields));
431  RCHECK(image->AddPixel(BitDepth::k4Bit, 0, is_top_fields));
432  } else if (switch_3 == 2) {
433  RCHECK(reader->ReadBits(4, &peek)); // run_length_9-24
434  uint8_t code;
435  RCHECK(reader->ReadBits(4, &code));
436  for (int i = 0; i < peek + 9; i++)
437  RCHECK(image->AddPixel(BitDepth::k4Bit, code, is_top_fields));
438  } else {
439  // switch_3 == 3
440  RCHECK(reader->ReadBits(8, &peek)); // run_length_25-280
441  uint8_t code;
442  RCHECK(reader->ReadBits(4, &code));
443  for (int i = 0; i < peek + 25; i++)
444  RCHECK(image->AddPixel(BitDepth::k4Bit, code, is_top_fields));
445  }
446  }
447  }
448  }
449  }
450  return true;
451 }

453 bool DvbSubParser::Parse8BitPixelData(bool is_top_fields,
454  BitReader* reader,
455  DvbImageBuilder* image) {
456  // 8-bit/pixel code string, Section 7.2.5.2.3, Table 26.
457  while (true) {
458  uint8_t peek;
459  RCHECK(reader->ReadBits(8, &peek));
460  if (peek != 0) {
461  RCHECK(image->AddPixel(BitDepth::k8Bit, peek, is_top_fields));
462  } else {
463  uint8_t switch_1;
464  RCHECK(reader->ReadBits(1, &switch_1));
465  if (switch_1 == 0) {
466  RCHECK(reader->ReadBits(7, &peek));
467  if (peek != 0) {
468  for (uint8_t i = 0; i < peek; i++)
469  RCHECK(image->AddPixel(BitDepth::k8Bit, 0, is_top_fields));
470  } else {
471  break;
472  }
473  } else {
474  uint8_t count;
475  RCHECK(reader->ReadBits(7, &count));
476  RCHECK(reader->ReadBits(8, &peek));
477  for (uint8_t i = 0; i < count; i++)
478  RCHECK(image->AddPixel(BitDepth::k8Bit, peek, is_top_fields));
479  }
480  }
481  }
482 
483  return true;
