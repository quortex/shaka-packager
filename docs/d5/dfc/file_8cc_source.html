78  const std::string dir_name = file_path.DirName().AsUTF8Unsafe();

138 const FileTypeInfo* GetFileTypeInfo(base::StringPiece file_name,
139  base::StringPiece* real_file_name) {
140  base::StringPiece file_type_prefix = GetFileTypePrefix(file_name);
141  for (const FileTypeInfo& file_type : kFileTypeInfo) {
142  if (file_type_prefix == file_type.type) {
143  *real_file_name = file_name.substr(file_type_prefix.size());
144  return &file_type;
145  }
146  }
147  // Otherwise we default to the first file type, which is LocalFile.
148  *real_file_name = file_name;
149  return &kFileTypeInfo[0];
150 }

154 File* File::Create(const char* file_name, const char* mode) {
155  std::unique_ptr<File, FileCloser> internal_file(
156  CreateInternalFile(file_name, mode));
157 
158  base::StringPiece file_type_prefix = GetFileTypePrefix(file_name);
159  if (file_type_prefix == kMemoryFilePrefix ||
160  file_type_prefix == kCallbackFilePrefix) {
161  // Disable caching for memory and callback files.
162  return internal_file.release();
163  }
164 
165  if (FLAGS_io_cache_size) {
166  // Enable threaded I/O for "r", "w", and "a" modes only.
167  if (!strcmp(mode, "r")) {
168  return new ThreadedIoFile(std::move(internal_file),
169  ThreadedIoFile::kInputMode, FLAGS_io_cache_size,
170  FLAGS_io_block_size);
171  } else if (!strcmp(mode, "w") || !strcmp(mode, "a")) {
172  return new ThreadedIoFile(std::move(internal_file),
173  ThreadedIoFile::kOutputMode,
174  FLAGS_io_cache_size, FLAGS_io_block_size);
175  }
176  }
177 
178  // Threaded I/O is disabled.
179  DLOG(WARNING) << "Threaded I/O is disabled. Performance may be decreased.";
180  return internal_file.release();
181 }

241 bool File::ReadFileToString(const char* file_name, std::string* contents) {
242  DCHECK(contents);
243 
244  File* file = File::Open(file_name, "r");
245  if (!file)
246  return false;
247 
248  const size_t kBufferSize = 0x40000; // 256KB.
249  std::unique_ptr<char[]> buf(new char[kBufferSize]);
250 
251  int64_t len;
252  while ((len = file->Read(buf.get(), kBufferSize)) > 0)
253  contents->append(buf.get(), len);
254 
255  file->Close();
256  return len == 0;
257 }

259 bool File::WriteStringToFile(const char* file_name,
260  const std::string& contents) {
261  VLOG(2) << "File::WriteStringToFile: " << file_name;
262  std::unique_ptr<File, FileCloser> file(File::Open(file_name, "w"));
263  if (!file) {
264  LOG(ERROR) << "Failed to open file " << file_name;
265  return false;
266  }
267  int64_t bytes_written = file->Write(contents.data(), contents.size());
268  if (bytes_written < 0) {
269  LOG(ERROR) << "Failed to write to file '" << file_name << "' ("
270  << bytes_written << ").";
271  return false;
272  }
273  if (static_cast<size_t>(bytes_written) != contents.size()) {
274  LOG(ERROR) << "Failed to write the whole file to " << file_name
275  << ". Wrote " << bytes_written << " but expecting "
276  << contents.size() << " bytes.";
277  return false;
278  }
279  if (!file.release()->Close()) {
280  LOG(ERROR)
281  << "Failed to close file '" << file_name
282  << "', possibly file permission issue or running out of disk space.";
283  return false;
284  }
285  return true;
286 }

288 bool File::WriteFileAtomically(const char* file_name,
289  const std::string& contents) {
290  VLOG(2) << "File::WriteFileAtomically: " << file_name;
291  base::StringPiece real_file_name;
292  const FileTypeInfo* file_type = GetFileTypeInfo(file_name, &real_file_name);
293  DCHECK(file_type);
294  if (file_type->atomic_write_function)
295  return file_type->atomic_write_function(real_file_name.data(), contents);
296 
297  // Provide a default implementation which may not be atomic unfortunately.
298 
299  // Skip the warning message for memory files, which is meant for testing
300  // anyway..
301  // Also check for http files, as they can't do atomic writes.
302  if (strncmp(file_name, kMemoryFilePrefix, strlen(kMemoryFilePrefix)) != 0
303  && strncmp(file_name, kHttpFilePrefix, strlen(kHttpFilePrefix)) != 0
304  && strncmp(file_name, kHttpsFilePrefix, strlen(kHttpsFilePrefix)) != 0) {
305  LOG(WARNING) << "Writing to " << file_name
306  << " is not guaranteed to be atomic.";
307  }
308  return WriteStringToFile(file_name, contents);
309 }

311 bool File::Copy(const char* from_file_name, const char* to_file_name) {
312  std::string content;
313  VLOG(2) << "File::Copy from " << from_file_name << " to " << to_file_name;
314  if (!ReadFileToString(from_file_name, &content)) {
315  LOG(ERROR) << "Failed to open file " << from_file_name;
316  return false;
317  }
318 
319  std::unique_ptr<File, FileCloser> output_file(File::Open(to_file_name, "w"));
320  if (!output_file) {
321  LOG(ERROR) << "Failed to write to " << to_file_name;
322  return false;
323  }
324 
325  uint64_t bytes_left = content.size();
326  uint64_t total_bytes_written = 0;
327  const char* content_cstr = content.c_str();
328  while (bytes_left > total_bytes_written) {
329  const int64_t bytes_written =
330  output_file->Write(content_cstr + total_bytes_written, bytes_left);
331  if (bytes_written < 0) {
332  LOG(ERROR) << "Failure while writing to " << to_file_name;
333  return false;
334  }
335 
336  total_bytes_written += bytes_written;
337  }
338  if (!output_file.release()->Close()) {
339  LOG(ERROR)
340  << "Failed to close file '" << to_file_name
341  << "', possibly file permission issue or running out of disk space.";
342  return false;
343  }
344  return true;
345 }

351 int64_t File::CopyFile(File* source, File* destination, int64_t max_copy) {
352  DCHECK(source);
353  DCHECK(destination);
354  if (max_copy < 0)
355  max_copy = std::numeric_limits<int64_t>::max();
356 
357  VLOG(2) << "File::CopyFile from " << source->file_name() << " to "
358  << destination->file_name();
359 
360  const int64_t kBufferSize = 0x40000; // 256KB.
361  std::unique_ptr<uint8_t[]> buffer(new uint8_t[kBufferSize]);
362  int64_t bytes_copied = 0;
363  while (bytes_copied < max_copy) {
364  const int64_t size = std::min(kBufferSize, max_copy - bytes_copied);
365  const int64_t bytes_read = source->Read(buffer.get(), size);
366  if (bytes_read < 0)
367  return bytes_read;
368  if (bytes_read == 0)
369  break;
370 
371  int64_t total_bytes_written = 0;
372  while (total_bytes_written < bytes_read) {
373  const int64_t bytes_written = destination->Write(
374  buffer.get() + total_bytes_written, bytes_read - total_bytes_written);
375  if (bytes_written < 0)
376  return bytes_written;
377 
378  total_bytes_written += bytes_written;
379  }
380 
381  DCHECK_EQ(total_bytes_written, bytes_read);
382  bytes_copied += bytes_read;
383  }
384 
385  return bytes_copied;
386 }
