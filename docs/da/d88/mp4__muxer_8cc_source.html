37 // Sets the range start and end value from offset and size.
38 // |start| and |end| are for byte-range-spec specified in RFC2616.
39 void SetStartAndEndFromOffsetAndSize(size_t offset,
40  size_t size,
41  Range* range) {
42  DCHECK(range);
43  range->start = static_cast<uint32_t>(offset);
44  // Note that ranges are inclusive. So we need - 1.
45  range->end = range->start + static_cast<uint32_t>(size) - 1;
46 }

48 FourCC CodecToFourCC(Codec codec, H26xStreamFormat h26x_stream_format) {
49  switch (codec) {
50  case kCodecAV1:
51  return FOURCC_av01;
52  case kCodecH264:
53  return h26x_stream_format ==
54  H26xStreamFormat::kNalUnitStreamWithParameterSetNalus
55  ? FOURCC_avc3
56  : FOURCC_avc1;
57  case kCodecH265:
58  return h26x_stream_format ==
59  H26xStreamFormat::kNalUnitStreamWithParameterSetNalus
60  ? FOURCC_hev1
61  : FOURCC_hvc1;
62  case kCodecH265DolbyVision:
63  return h26x_stream_format ==
64  H26xStreamFormat::kNalUnitStreamWithParameterSetNalus
65  ? FOURCC_dvhe
66  : FOURCC_dvh1;
67  case kCodecVP8:
68  return FOURCC_vp08;
69  case kCodecVP9:
70  return FOURCC_vp09;
71  case kCodecAAC:
72  case kCodecMP3:
73  return FOURCC_mp4a;
74  case kCodecAC3:
75  return FOURCC_ac_3;
76  case kCodecDTSC:
77  return FOURCC_dtsc;
78  case kCodecDTSH:
79  return FOURCC_dtsh;
80  case kCodecDTSL:
81  return FOURCC_dtsl;
82  case kCodecDTSE:
83  return FOURCC_dtse;
84  case kCodecDTSM:
85  return FOURCC_dtsm;
86  case kCodecEAC3:
87  return FOURCC_ec_3;
88  case kCodecAC4:
89  return FOURCC_ac_4;
90  case kCodecFlac:
91  return FOURCC_fLaC;
92  case kCodecOpus:
93  return FOURCC_Opus;
94  case kCodecMha1:
95  return FOURCC_mha1;
96  case kCodecMhm1:
97  return FOURCC_mhm1;
98  default:
99  return FOURCC_NULL;
100  }

103 void GenerateSinf(FourCC old_type,
104  const EncryptionConfig& encryption_config,
105  ProtectionSchemeInfo* sinf) {
106  sinf->format.format = old_type;
107 
108  DCHECK_NE(encryption_config.protection_scheme, FOURCC_NULL);
109  sinf->type.type = encryption_config.protection_scheme;
110 
111  // The version of cenc implemented here. CENC 4.
112  const int kCencSchemeVersion = 0x00010000;
113  sinf->type.version = kCencSchemeVersion;
114 
115  auto& track_encryption = sinf->info.track_encryption;
116  track_encryption.default_is_protected = 1;
117 
118  track_encryption.default_crypt_byte_block =
119  encryption_config.crypt_byte_block;
120  track_encryption.default_skip_byte_block = encryption_config.skip_byte_block;
121  switch (encryption_config.protection_scheme) {
122  case FOURCC_cenc:
123  case FOURCC_cbc1:
124  DCHECK_EQ(track_encryption.default_crypt_byte_block, 0u);
125  DCHECK_EQ(track_encryption.default_skip_byte_block, 0u);
126  // CENCv3 10.1 ‘cenc’ AES-CTR scheme and 10.2 ‘cbc1’ AES-CBC scheme:
127  // The version of the Track Encryption Box (‘tenc’) SHALL be 0.
128  track_encryption.version = 0;
129  break;
130  case FOURCC_cbcs:
131  case FOURCC_cens:
132  // CENCv3 10.3 ‘cens’ AES-CTR subsample pattern encryption scheme and
133  // 10.4 ‘cbcs’ AES-CBC subsample pattern encryption scheme:
134  // The version of the Track Encryption Box (‘tenc’) SHALL be 1.
135  track_encryption.version = 1;
136  break;
137  default:
138  NOTREACHED() << "Unexpected protection scheme "
139  << encryption_config.protection_scheme;
140  }
141 
142  track_encryption.default_per_sample_iv_size =
143  encryption_config.per_sample_iv_size;
144  track_encryption.default_constant_iv = encryption_config.constant_iv;
145  track_encryption.default_kid = encryption_config.key_id;
146 }

148 // The roll distance is expressed in sample units and always takes negative
149 // values.
150 int16_t GetRollDistance(uint64_t seek_preroll_ns, uint32_t sampling_frequency) {
151  const double kNanosecondsPerSecond = 1000000000;
152  const double preroll_in_samples =
153  seek_preroll_ns / kNanosecondsPerSecond * sampling_frequency;
154  // Round to closest integer.
155  return -static_cast<int16_t>(preroll_in_samples + 0.5);
156 }

169 Status MP4Muxer::Finalize() {
170  // This happens on streams that are not initialized, i.e. not going through
171  // DelayInitializeMuxer, which can only happen if there are no samples from
172  // the stream.
173  if (!segmenter_) {
174  DCHECK(to_be_initialized_);
175  LOG(INFO) << "Skip stream '" << options().output_file_name
176  << "' which does not contain any sample.";
177  return Status::OK;
178  }
179 
180  Status segmenter_finalized = segmenter_->Finalize();
181 
182  if (!segmenter_finalized.ok())
183  return segmenter_finalized;
184 
185  FireOnMediaEndEvent();
186  LOG(INFO) << "MP4 file '" << options().output_file_name << "' finalized.";
187  return Status::OK;
188 }

209 Status MP4Muxer::DelayInitializeMuxer() {
210  DCHECK(!streams().empty());
211 
212  std::unique_ptr<FileType> ftyp(new FileType);
213  std::unique_ptr<Movie> moov(new Movie);
214 
215  ftyp->major_brand = FOURCC_mp41;
216  ftyp->compatible_brands.push_back(FOURCC_iso8);
217  ftyp->compatible_brands.push_back(FOURCC_isom);
218  ftyp->compatible_brands.push_back(FOURCC_mp41);
219  ftyp->compatible_brands.push_back(FOURCC_dash);
220 
221  if (streams().size() == 1) {
222  FourCC codec_fourcc = FOURCC_NULL;
223  if (streams()[0]->stream_type() == kStreamVideo) {
224  codec_fourcc =
225  CodecToFourCC(streams()[0]->codec(),
226  static_cast<const VideoStreamInfo*>(streams()[0].get())
227  ->h26x_stream_format());
228  if (codec_fourcc != FOURCC_NULL)
229  ftyp->compatible_brands.push_back(codec_fourcc);
230  }
231 
232  // CMAF allows only one track/stream per file.
233  // CMAF requires single initialization switching for AVC3/HEV1, which is not
234  // supported yet.
235  if (codec_fourcc != FOURCC_avc3 && codec_fourcc != FOURCC_hev1)
236  ftyp->compatible_brands.push_back(FOURCC_cmfc);
237  }
238 
239  moov->header.creation_time = IsoTimeNow();
240  moov->header.modification_time = IsoTimeNow();
241  moov->header.next_track_id = static_cast<uint32_t>(streams().size()) + 1;
242 
243  moov->tracks.resize(streams().size());
244  moov->extends.tracks.resize(streams().size());
245 
246  // Initialize tracks.
247  for (uint32_t i = 0; i < streams().size(); ++i) {
248  const StreamInfo* stream = streams()[i].get();
249  Track& trak = moov->tracks[i];
250  trak.header.track_id = i + 1;
251 
252  TrackExtends& trex = moov->extends.tracks[i];
253  trex.track_id = trak.header.track_id;
254  trex.default_sample_description_index = 1;
255 
256  bool generate_trak_result = false;
257  switch (stream->stream_type()) {
258  case kStreamVideo:
259  generate_trak_result = GenerateVideoTrak(
260  static_cast<const VideoStreamInfo*>(stream), &trak);
261  break;
262  case kStreamAudio:
263  generate_trak_result = GenerateAudioTrak(
264  static_cast<const AudioStreamInfo*>(stream), &trak);
265  break;
266  case kStreamText:
267  generate_trak_result = GenerateTextTrak(
268  static_cast<const TextStreamInfo*>(stream), &trak);
269  break;
270  default:
271  NOTIMPLEMENTED() << "Not implemented for stream type: "
272  << stream->stream_type();
273  }
274  if (!generate_trak_result)
275  return Status(error::MUXER_FAILURE, "Failed to generate trak.");
276 
277  // Generate EditList if needed. See UpdateEditListOffsetFromSample() for
278  // more information.
279  if (edit_list_offset_.value() > 0) {
280  EditListEntry entry;
281  entry.media_time = edit_list_offset_.value();
282  entry.media_rate_integer = 1;
283  trak.edit.list.edits.push_back(entry);
284  }
285 
286  if (stream->is_encrypted() && options().mp4_params.include_pssh_in_stream) {
287  moov->pssh.clear();
288  const auto& key_system_info = stream->encryption_config().key_system_info;
289  for (const ProtectionSystemSpecificInfo& system : key_system_info) {
290  if (system.psshs.empty())
291  continue;
292  ProtectionSystemSpecificHeader pssh;
293  pssh.raw_box = system.psshs;
294  moov->pssh.push_back(pssh);
295  }
296  }
297  }
298 
299  if (options().segment_template.empty()) {
300  segmenter_.reset(new SingleSegmentSegmenter(options(), std::move(ftyp),
301  std::move(moov)));
302  } else if (options().mp4_params.low_latency_dash_mode) {
303  segmenter_.reset(new LowLatencySegmentSegmenter(options(), std::move(ftyp),
304  std::move(moov)));
305  } else {
306  segmenter_.reset(
307  new MultiSegmentSegmenter(options(), std::move(ftyp), std::move(moov)));
308  }
309 
310  const Status segmenter_initialized =
311  segmenter_->Initialize(streams(), muxer_listener(), progress_listener());
312  if (!segmenter_initialized.ok())
313  return segmenter_initialized;
314 
315  FireOnMediaStartEvent();
316  return Status::OK;
317 }

319 Status MP4Muxer::UpdateEditListOffsetFromSample(const MediaSample& sample) {
320  if (edit_list_offset_)
321  return Status::OK;
322 
323  const int64_t pts = sample.pts();
324  const int64_t dts = sample.dts();
325  // An EditList entry is inserted if one of the below conditions occur [4]:
326  // (1) pts > dts for the first sample. Due to Chrome's dts bug [1], dts is
327  // used in buffered range API, while pts is used elsewhere (players,
328  // manifests, and Chrome's own appendWindow check etc.), this
329  // inconsistency creates various problems, including possible stalls
330  // during playback. Since Chrome adjusts pts only when seeing EditList
331  // [2], we can insert an EditList with the time equal to difference of pts
332  // and dts to make aligned buffered ranges using pts and dts. This
333  // effectively workarounds the dts bug. It is also recommended by ISO-BMFF
334  // specification [3].
335  // (2) pts == dts and with pts < 0. This happens for some audio codecs where a
336  // negative presentation timestamp signals that the sample is not supposed
337  // to be shown, i.e. for audio priming. EditList is needed to encode
338  // negative timestamps.
339  // [1] https://crbug.com/718641, fixed but behind MseBufferByPts, still not
340  // enabled as of M67.
341  // [2] This is actually a bug, see https://crbug.com/354518. It looks like
342  // Chrome is planning to enable the fix for [1] before addressing this
343  // bug, so we are safe.
344  // [3] ISO 14496-12:2015 8.6.6.1
345  // It is recommended that such an edit be used to establish a presentation
346  // time of 0 for the first presented sample, when composition offsets are
347  // used.
348  // [4] ISO 23009-19:2018 7.5.13
349  // In two cases, an EditBox containing a single EditListBox with the
350  // following constraints may be present in the CMAF header of a CMAF track
351  // to adjust the presentation time of all media samples in the CMAF track.
352  // a) The first case is a video CMAF track file using v0 TrackRunBoxes
353  // with positive composition offsets to reorder video media samples.
354  // b) The second case is an audio CMAF track where each media sample's
355  // presentation time does not equal its composition time.
356  const int64_t pts_dts_offset = pts - dts;
357  if (pts_dts_offset > 0) {
358  if (pts < 0) {
359  LOG(ERROR) << "Negative presentation timestamp (" << pts
360  << ") is not supported when there is an offset between "
361  "presentation timestamp and decoding timestamp ("
362  << dts << ").";
363  return Status(error::MUXER_FAILURE,
364  "Unsupported negative pts when there is an offset between "
365  "pts and dts.");
366  }
367  edit_list_offset_ = pts_dts_offset;
368  return Status::OK;
369  }
370  if (pts_dts_offset < 0) {
371  LOG(ERROR) << "presentation timestamp (" << pts
372  << ") is not supposed to be greater than decoding timestamp ("
373  << dts << ").";
374  return Status(error::MUXER_FAILURE, "Not expecting pts < dts.");
375  }
376  edit_list_offset_ = std::max(-sample.pts(), static_cast<int64_t>(0));
377  return Status::OK;
378 }

380 void MP4Muxer::InitializeTrak(const StreamInfo* info, Track* trak) {
381  int64_t now = IsoTimeNow();
382  trak->header.creation_time = now;
383  trak->header.modification_time = now;
384  trak->header.duration = 0;
385  trak->media.header.creation_time = now;
386  trak->media.header.modification_time = now;
387  trak->media.header.timescale = info->time_scale();
388  trak->media.header.duration = 0;
389  if (!info->language().empty()) {
390  // Strip off the subtag, if any.
391  std::string main_language = info->language();
392  size_t dash = main_language.find('-');
393  if (dash != std::string::npos) {
394  main_language.erase(dash);
395  }
396 
397  // ISO-639-2/T main language code should be 3 characters.
398  if (main_language.size() != 3) {
399  LOG(WARNING) << "'" << main_language << "' is not a valid ISO-639-2 "
400  << "language code, ignoring.";
401  } else {
402  trak->media.header.language.code = main_language;
403  }
404  }
405 }

407 bool MP4Muxer::GenerateVideoTrak(const VideoStreamInfo* video_info,
408  Track* trak) {
409  InitializeTrak(video_info, trak);
410 
411  // width and height specify the track's visual presentation size as
412  // fixed-point 16.16 values.
413  uint32_t pixel_width = video_info->pixel_width();
414  uint32_t pixel_height = video_info->pixel_height();
415  if (pixel_width == 0 || pixel_height == 0) {
416  LOG(WARNING) << "pixel width/height are not set. Assuming 1:1.";
417  pixel_width = 1;
418  pixel_height = 1;
419  }
420  const double sample_aspect_ratio =
421  static_cast<double>(pixel_width) / pixel_height;
422  trak->header.width = video_info->width() * sample_aspect_ratio * 0x10000;
423  trak->header.height = video_info->height() * 0x10000;
424 
425  VideoSampleEntry video;
426  video.format =
427  CodecToFourCC(video_info->codec(), video_info->h26x_stream_format());
428  video.width = video_info->width();
429  video.height = video_info->height();
430  video.codec_configuration.data = video_info->codec_config();
431  if (!video.ParseExtraCodecConfigsVector(video_info->extra_config())) {
432  LOG(ERROR) << "Malformed extra codec configs: "
433  << base::HexEncode(video_info->extra_config().data(),
434  video_info->extra_config().size());
435  return false;
436  }
437  if (pixel_width != 1 || pixel_height != 1) {
438  video.pixel_aspect.h_spacing = pixel_width;
439  video.pixel_aspect.v_spacing = pixel_height;
440  }
441 
442  SampleDescription& sample_description =
443  trak->media.information.sample_table.description;
444  sample_description.type = kVideo;
445  sample_description.video_entries.push_back(video);
446 
447  if (video_info->is_encrypted()) {
448  if (video_info->has_clear_lead()) {
449  // Add a second entry for clear content.
450  sample_description.video_entries.push_back(video);
451  }
452  // Convert the first entry to an encrypted entry.
453  VideoSampleEntry& entry = sample_description.video_entries[0];
454  GenerateSinf(entry.format, video_info->encryption_config(), &entry.sinf);
455  entry.format = FOURCC_encv;
456  }
457  return true;
458 }

460 bool MP4Muxer::GenerateAudioTrak(const AudioStreamInfo* audio_info,
461  Track* trak) {
462  InitializeTrak(audio_info, trak);
463 
464  trak->header.volume = 0x100;
465 
466  AudioSampleEntry audio;
467  audio.format =
468  CodecToFourCC(audio_info->codec(), H26xStreamFormat::kUnSpecified);
469  switch(audio_info->codec()){
470  case kCodecAAC: {
471  DecoderConfigDescriptor* decoder_config =
472  audio.esds.es_descriptor.mutable_decoder_config_descriptor();
473  decoder_config->set_object_type(ObjectType::kISO_14496_3); // MPEG4 AAC.
474  decoder_config->set_max_bitrate(audio_info->max_bitrate());
475  decoder_config->set_avg_bitrate(audio_info->avg_bitrate());
476  decoder_config->mutable_decoder_specific_info_descriptor()->set_data(
477  audio_info->codec_config());
478  break;
479  }
480  case kCodecDTSC:
481  case kCodecDTSH:
482  case kCodecDTSL:
483  case kCodecDTSE:
484  case kCodecDTSM:
485  audio.ddts.extra_data = audio_info->codec_config();
486  audio.ddts.max_bitrate = audio_info->max_bitrate();
487  audio.ddts.avg_bitrate = audio_info->avg_bitrate();
488  audio.ddts.sampling_frequency = audio_info->sampling_frequency();
489  audio.ddts.pcm_sample_depth = audio_info->sample_bits();
490  break;
491  case kCodecAC3:
492  audio.dac3.data = audio_info->codec_config();
493  break;
494  case kCodecEAC3:
495  audio.dec3.data = audio_info->codec_config();
496  break;
497  case kCodecAC4:
498  audio.dac4.data = audio_info->codec_config();
499  break;
500  case kCodecFlac:
501  audio.dfla.data = audio_info->codec_config();
502  break;
503  case kCodecMP3: {
504  DecoderConfigDescriptor* decoder_config =
505  audio.esds.es_descriptor.mutable_decoder_config_descriptor();
506  uint32_t samplerate = audio_info->sampling_frequency();
507  if (samplerate < 32000)
508  decoder_config->set_object_type(ObjectType::kISO_13818_3_MPEG1);
509  else
510  decoder_config->set_object_type(ObjectType::kISO_11172_3_MPEG1);
511  decoder_config->set_max_bitrate(audio_info->max_bitrate());
512  decoder_config->set_avg_bitrate(audio_info->avg_bitrate());
513 
514  // For values of DecoderConfigDescriptor.objectTypeIndication
515  // that refer to streams complying with ISO/IEC 11172-3 or
516  // ISO/IEC 13818-3 the decoder specific information is empty
517  // since all necessary data is contained in the bitstream frames
518  // itself.
519  break;
520  }
521  case kCodecOpus:
522  audio.dops.opus_identification_header = audio_info->codec_config();
523  break;
524  case kCodecMha1:
525  case kCodecMhm1:
526  audio.mhac.data = audio_info->codec_config();
527  break;
528  default:
529  NOTIMPLEMENTED() << " Unsupported audio codec " << audio_info->codec();
530  return false;
531  }
532 
533  if (audio_info->codec() == kCodecAC3 || audio_info->codec() == kCodecEAC3) {
534  // AC3 and EC3 does not fill in actual channel count and sample size in
535  // sample description entry. Instead, two constants are used.
536  audio.channelcount = 2;
537  audio.samplesize = 16;
538  } else if (audio_info->codec() == kCodecAC4) {
539  //ETSI TS 103 190-2, E.4.5 channelcount should be set to the total number of
540  //audio outputchannels of the default audio presentation of that track
541  audio.channelcount = audio_info->num_channels();
542  //ETSI TS 103 190-2, E.4.6 samplesize shall be set to 16.
543  audio.samplesize = 16;
544  } else {
545  audio.channelcount = audio_info->num_channels();
546  audio.samplesize = audio_info->sample_bits();
547  }
548  audio.samplerate = audio_info->sampling_frequency();
549  SampleTable& sample_table = trak->media.information.sample_table;
550  SampleDescription& sample_description = sample_table.description;
551  sample_description.type = kAudio;
552  sample_description.audio_entries.push_back(audio);
553 
554  if (audio_info->is_encrypted()) {
555  if (audio_info->has_clear_lead()) {
556  // Add a second entry for clear content.
557  sample_description.audio_entries.push_back(audio);
558  }
559  // Convert the first entry to an encrypted entry.
560  AudioSampleEntry& entry = sample_description.audio_entries[0];
561  GenerateSinf(entry.format, audio_info->encryption_config(), &entry.sinf);
562  entry.format = FOURCC_enca;
563  }
564 
565  if (audio_info->seek_preroll_ns() > 0) {
566  sample_table.sample_group_descriptions.resize(1);
567  SampleGroupDescription& sample_group_description =
568  sample_table.sample_group_descriptions.back();
569  sample_group_description.grouping_type = FOURCC_roll;
570  sample_group_description.audio_roll_recovery_entries.resize(1);
571  sample_group_description.audio_roll_recovery_entries[0].roll_distance =

607  sample_description.text_entries.push_back(webvtt);
608  return true;
