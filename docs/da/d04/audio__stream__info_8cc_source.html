132 std::string AudioStreamInfo::GetCodecString(Codec codec,
133  uint8_t audio_object_type) {
134  switch (codec) {
135  case kCodecAAC:
136  return "mp4a.40." + base::UintToString(audio_object_type);
137  case kCodecAC3:
138  return "ac-3";
139  case kCodecDTSC:
140  return "dtsc";
141  case kCodecDTSE:
142  return "dtse";
143  case kCodecDTSH:
144  return "dtsh";
145  case kCodecDTSL:
146  return "dtsl";
147  case kCodecDTSM:
148  return "dts-";
149  case kCodecDTSP:
150  return "dts+";
151  case kCodecEAC3:
152  return "ec-3";
153  case kCodecAC4:
154  // ETSI TS 103 190-2 Digital Audio Compression (AC-4) Standard; Part 2:
155  // Immersive and personalized audio E.13. audio_object_type is composed of
156  // bitstream_version (3bits), presentation_version (2bits) and
157  // mdcompat (3bits).
158  return base::StringPrintf(
159  "ac-4.%02d.%02d.%02d", (audio_object_type & 0xE0) >> 5,
160  (audio_object_type & 0x18) >> 3, audio_object_type & 0x7);
161  case kCodecFlac:
162  return "flac";
163  case kCodecOpus:
164  return "opus";
165  case kCodecMP3:
166  return "mp3";
167  case kCodecVorbis:
168  return "vorbis";
169  case kCodecMha1:
170  case kCodecMhm1:
171  // The signalling of the codecs parameters is according to RFC6381 [11]
172  // and ISO/IEC 23008-3 clause 21 [7].
173  // The value consists of the following two parts separated by a dot:
174  // - the sample entry 4CC code ('mha1', 'mha2', 'mhm1', 'mhm2')
175  // - ‘0x’ followed by the hex value of the profile-levelid, as defined
176  // in in ISO/IEC 23008-3 [7]
177  return base::StringPrintf("%s.0x%02x",
178  FourCCToString(CodecToFourCC(codec)).c_str(),
179  audio_object_type);
180  default:
181  NOTIMPLEMENTED() << "Codec: " << codec;
182  return "unknown";
183  }
184 }
