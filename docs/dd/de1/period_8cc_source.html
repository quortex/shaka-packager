46  if (role_str == "caption")

48  if (role_str == "subtitle")

74 AdaptationSet* Period::GetOrCreateAdaptationSet(
75  const MediaInfo& media_info,
76  bool content_protection_in_adaptation_set) {
77  // Set duration if it is not set. It may be updated later from duration
78  // calculated from segments.
79  if (duration_seconds_ == 0)
80  duration_seconds_ = media_info.media_duration_seconds();
81 
82  const std::string key = GetAdaptationSetKey(
83  media_info, mpd_options_.mpd_params.allow_codec_switching);
84 
85  std::list<AdaptationSet*>& adaptation_sets = adaptation_set_list_map_[key];
86 
87  for (AdaptationSet* adaptation_set : adaptation_sets) {
88  if (protected_adaptation_set_map_.Match(
89  *adaptation_set, media_info, content_protection_in_adaptation_set))
90  return adaptation_set;
91  }
92 
93  // None of the adaptation sets match with the new content protection.
94  // Need a new one.
95  const std::string language = GetLanguage(media_info);
96  std::unique_ptr<AdaptationSet> new_adaptation_set =
97  NewAdaptationSet(language, mpd_options_, representation_counter_);
98  if (!SetNewAdaptationSetAttributes(language, media_info, adaptation_sets,
99  content_protection_in_adaptation_set,
100  new_adaptation_set.get())) {
101  return nullptr;
102  }
103 
104  if (content_protection_in_adaptation_set &&
105  media_info.has_protected_content()) {
106  protected_adaptation_set_map_.Register(*new_adaptation_set, media_info);
107  AddContentProtectionElements(media_info, new_adaptation_set.get());
108  }
109  for (AdaptationSet* adaptation_set : adaptation_sets) {
110  if (protected_adaptation_set_map_.Switchable(*adaptation_set,
111  *new_adaptation_set)) {
112  adaptation_set->AddAdaptationSetSwitching(new_adaptation_set.get());
113  new_adaptation_set->AddAdaptationSetSwitching(adaptation_set);
114  }
115  }
116 
117  AdaptationSet* adaptation_set_ptr = new_adaptation_set.get();
118  adaptation_sets.push_back(adaptation_set_ptr);
119  adaptation_sets_.emplace_back(std::move(new_adaptation_set));
120  return adaptation_set_ptr;
121 }

123 base::Optional<xml::XmlNode> Period::GetXml(bool output_period_duration) {
124  adaptation_sets_.sort(
125  [](const std::unique_ptr<AdaptationSet>& adaptation_set_a,
126  const std::unique_ptr<AdaptationSet>& adaptation_set_b) {
127  if (!adaptation_set_a->has_id())
128  return false;
129  if (!adaptation_set_b->has_id())
130  return true;
131  return adaptation_set_a->id() < adaptation_set_b->id();
132  });
133 
134  xml::XmlNode period("Period");
135 
136  // Required for 'dynamic' MPDs.
137  if (!period.SetId(id_))
138  return base::nullopt;
139 
140  // Required for LL-DASH MPDs.
141  if (mpd_options_.mpd_params.low_latency_dash_mode) {
142  // Create ServiceDescription element.
143  xml::XmlNode service_description_node("ServiceDescription");
144  if (!service_description_node.SetIntegerAttribute("id", id_))
145  return base::nullopt;
146 
147  // Insert Latency into ServiceDescription element.
148  xml::XmlNode latency_node("Latency");
149  uint64_t target_latency_ms =
150  mpd_options_.mpd_params.target_latency_seconds * 1000;
151  if (!latency_node.SetIntegerAttribute("target", target_latency_ms))
152  return base::nullopt;
153  if (!service_description_node.AddChild(std::move(latency_node)))
154  return base::nullopt;
155 
156  // Insert ServiceDescription into Period element.
157  if (!period.AddChild(std::move(service_description_node)))
158  return base::nullopt;
159  }
160 
161  // Iterate thru AdaptationSets and add them to one big Period element.
162  for (const auto& adaptation_set : adaptation_sets_) {
163  auto child = adaptation_set->GetXml();
164  if (!child || !period.AddChild(std::move(*child)))
165  return base::nullopt;
166  }
167 
168  if (output_period_duration) {
169  if (!period.SetStringAttribute("duration",
170  SecondsToXmlDuration(duration_seconds_))) {
171  return base::nullopt;
172  }
173  } else if (mpd_options_.mpd_type == MpdType::kDynamic) {
174  if (!period.SetStringAttribute(
175  "start", SecondsToXmlDuration(start_time_in_seconds_))) {
176  return base::nullopt;
177  }
178  }
179  return period;
180 }

198 bool Period::SetNewAdaptationSetAttributes(
199  const std::string& language,
200  const MediaInfo& media_info,
201  const std::list<AdaptationSet*>& adaptation_sets,
202  bool content_protection_in_adaptation_set,
203  AdaptationSet* new_adaptation_set) {
204  if (!media_info.dash_roles().empty()) {
205  for (const std::string& role_str : media_info.dash_roles()) {
206  AdaptationSet::Role role = RoleFromString(role_str);
207  if (role == AdaptationSet::kRoleUnknown) {
208  LOG(ERROR) << "Unrecognized role '" << role_str << "'.";
209  return false;
210  }
211  new_adaptation_set->AddRole(role);
212  }
213  } else if (!language.empty()) {
214  const bool is_main_role =
215  language == (media_info.has_audio_info()
216  ? GetDefaultAudioLanguage(mpd_options_)
217  : GetDefaultTextLanguage(mpd_options_));
218  if (is_main_role)
219  new_adaptation_set->AddRole(AdaptationSet::kRoleMain);
220  }
221  for (const std::string& accessibility : media_info.dash_accessibilities()) {
222  size_t pos = accessibility.find('=');
223  if (pos == std::string::npos) {
224  LOG(ERROR)
225  << "Accessibility should be in scheme=value format, but seeing "
226  << accessibility;
227  return false;
228  }
229  new_adaptation_set->AddAccessibility(accessibility.substr(0, pos),
230  accessibility.substr(pos + 1));
231  }
232 
233  new_adaptation_set->set_codec(GetBaseCodec(media_info));
234 
235  if (media_info.has_video_info()) {
236  // Because 'language' is ignored for videos, |adaptation_sets| must have
237  // all the video AdaptationSets.
238  if (adaptation_sets.size() > 1) {
239  new_adaptation_set->AddRole(AdaptationSet::kRoleMain);
240  } else if (adaptation_sets.size() == 1) {
241  (*adaptation_sets.begin())->AddRole(AdaptationSet::kRoleMain);
242  new_adaptation_set->AddRole(AdaptationSet::kRoleMain);
243  }
244 
245  if (media_info.video_info().has_playback_rate()) {
246  std::string trick_play_reference_adaptation_set_key;
247  AdaptationSet* trick_play_reference_adaptation_set =
248  FindMatchingAdaptationSetForTrickPlay(
249  media_info, content_protection_in_adaptation_set,
250  &trick_play_reference_adaptation_set_key);
251  if (trick_play_reference_adaptation_set) {
252  new_adaptation_set->AddTrickPlayReference(
253  trick_play_reference_adaptation_set);
254  } else {
255  trickplay_cache_[trick_play_reference_adaptation_set_key].push_back(
256  new_adaptation_set);
257  }
258  } else {
259  std::string trick_play_adaptation_set_key;
260  AdaptationSet* trickplay_adaptation_set =
261  FindMatchingAdaptationSetForTrickPlay(
262  media_info, content_protection_in_adaptation_set,
263  &trick_play_adaptation_set_key);
264  if (trickplay_adaptation_set) {
265  trickplay_adaptation_set->AddTrickPlayReference(new_adaptation_set);
266  trickplay_cache_.erase(trick_play_adaptation_set_key);
267  }
268  }
269 
270  } else if (media_info.has_text_info()) {
271  // IOP requires all AdaptationSets to have (sub)segmentAlignment set to
272  // true, so carelessly set it to true.
273  // In practice it doesn't really make sense to adapt between text tracks.
274  new_adaptation_set->ForceSetSegmentAlignment(true);
275  }
276  return true;
277 }

279 AdaptationSet* Period::FindMatchingAdaptationSetForTrickPlay(
280  const MediaInfo& media_info,
281  bool content_protection_in_adaptation_set,
282  std::string* adaptation_set_key) {
283  std::list<AdaptationSet*>* adaptation_sets = nullptr;
284  const bool is_trickplay_adaptation_set =
285  media_info.video_info().has_playback_rate();
286  if (is_trickplay_adaptation_set) {
287  *adaptation_set_key = GetAdaptationSetKeyForTrickPlay(media_info);
288  if (adaptation_set_list_map_.find(*adaptation_set_key) ==
289  adaptation_set_list_map_.end())
290  return nullptr;
291  adaptation_sets = &adaptation_set_list_map_[*adaptation_set_key];
292  } else {
293  *adaptation_set_key = GetAdaptationSetKey(
294  media_info, mpd_options_.mpd_params.allow_codec_switching);
295  if (trickplay_cache_.find(*adaptation_set_key) == trickplay_cache_.end())
296  return nullptr;
297  adaptation_sets = &trickplay_cache_[*adaptation_set_key];
298  }
299  for (AdaptationSet* adaptation_set : *adaptation_sets) {
300  if (protected_adaptation_set_map_.Match(
301  *adaptation_set, media_info,
302  content_protection_in_adaptation_set)) {
303  return adaptation_set;
304  }
305  }
306 
307  return nullptr;
308 }

325 bool Period::ProtectedAdaptationSetMap::Match(
326  const AdaptationSet& adaptation_set,
327  const MediaInfo& media_info,
328  bool content_protection_in_adaptation_set) {
329  if (adaptation_set.codec() != GetBaseCodec(media_info))
330  return false;
331 
332  if (!content_protection_in_adaptation_set)
333  return true;
334 
335  const auto protected_content_it =
336  protected_content_map_.find(&adaptation_set);
337  // If the AdaptationSet ID is not registered in the map, then it is clear
338  // content.
339  if (protected_content_it == protected_content_map_.end())
340  return !media_info.has_protected_content();
341  if (!media_info.has_protected_content())
342  return false;
343 
344  return ProtectedContentEq(protected_content_it->second,
345  media_info.protected_content());
346 }

348 bool Period::ProtectedAdaptationSetMap::Switchable(
349  const AdaptationSet& adaptation_set_a,
350  const AdaptationSet& adaptation_set_b) {
351  const auto protected_content_it_a =
352  protected_content_map_.find(&adaptation_set_a);
353  const auto protected_content_it_b =
354  protected_content_map_.find(&adaptation_set_b);
355 
356  if (protected_content_it_a == protected_content_map_.end())
357  return protected_content_it_b == protected_content_map_.end();
358  if (protected_content_it_b == protected_content_map_.end())
359  return false;
360  // Get all the UUIDs of the AdaptationSet. If another AdaptationSet has the
361  // same UUIDs then those are switchable.
362  return GetUUIDs(protected_content_it_a->second) ==
363  GetUUIDs(protected_content_it_b->second);
364 }
