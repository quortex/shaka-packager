175 const inline std::string FourCCToString(FourCC fourcc) {
176  char buf[5];
177  buf[0] = (fourcc >> 24) & 0xff;
178  buf[1] = (fourcc >> 16) & 0xff;
179  buf[2] = (fourcc >> 8) & 0xff;
180  buf[3] = (fourcc) & 0xff;
181  buf[4] = 0;
182  return std::string(buf);
183 }
