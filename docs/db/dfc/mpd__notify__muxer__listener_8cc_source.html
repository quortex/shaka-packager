31 void MpdNotifyMuxerListener::OnEncryptionInfoReady(
32  bool is_initial_encryption_info,
33  FourCC protection_scheme,
34  const std::vector<uint8_t>& key_id,
35  const std::vector<uint8_t>& iv,
36  const std::vector<ProtectionSystemSpecificInfo>& key_system_info) {
37  if (is_initial_encryption_info) {
38  LOG_IF(WARNING, is_encrypted_)
39  << "Updating initial encryption information.";
40  protection_scheme_ = protection_scheme;
41  default_key_id_ = key_id;
42  key_system_info_ = key_system_info;
43  is_encrypted_ = true;
44  return;
45  }
46  if (!notification_id_)
47  return;
48  DCHECK_EQ(protection_scheme, protection_scheme_);
49 
50  for (const ProtectionSystemSpecificInfo& info : key_system_info) {
51  const std::string drm_uuid = internal::CreateUUIDString(info.system_id);
52  bool updated = mpd_notifier_->NotifyEncryptionUpdate(
53  notification_id_.value(), drm_uuid, key_id, info.psshs);
54  LOG_IF(WARNING, !updated) << "Failed to update encryption info.";
55  }
56 }

60 void MpdNotifyMuxerListener::OnMediaStart(const MuxerOptions& muxer_options,
61  const StreamInfo& stream_info,
62  int32_t time_scale,
63  ContainerType container_type) {
64  std::unique_ptr<MediaInfo> media_info(new MediaInfo());
65  if (!internal::GenerateMediaInfo(muxer_options,
66  stream_info,
67  time_scale,
68  container_type,
69  media_info.get())) {
70  LOG(ERROR) << "Failed to generate MediaInfo from input.";
71  return;
72  }
73  for (const std::string& accessibility : accessibilities_)
74  media_info->add_dash_accessibilities(accessibility);
75  if (roles_.empty() && stream_info.stream_type() == kStreamText) {
76  // If there aren't any roles, default to "subtitle" since some apps may
77  // require it to distinguish between subtitle/caption.
78  media_info->add_dash_roles("subtitle");
79  } else {
80  for (const std::string& role : roles_)
81  media_info->add_dash_roles(role);
82  }
83 
84  if (is_encrypted_) {
85  internal::SetContentProtectionFields(protection_scheme_, default_key_id_,
86  key_system_info_, media_info.get());
87  media_info->mutable_protected_content()->set_include_mspr_pro(mpd_notifier_->include_mspr_pro());
88  }
89 
90  // The content may be splitted into multiple files, but their MediaInfo
91  // should be compatible.
92  if (media_info_ &&
93  !internal::IsMediaInfoCompatible(*media_info, *media_info_)) {
94  LOG(WARNING) << "Incompatible MediaInfo \n"
95  << media_info->ShortDebugString() << "\n vs \n"
96  << media_info_->ShortDebugString()
97  << "\nThe result manifest may not be playable.";
98  }
99  media_info_ = std::move(media_info);
100 
101  if (mpd_notifier_->dash_profile() == DashProfile::kLive) {
102  if (!NotifyNewContainer())
103  return;
104  DCHECK(notification_id_);
105  }
106 }

140 void MpdNotifyMuxerListener::OnMediaEnd(const MediaRanges& media_ranges,
141  float duration_seconds) {
142  if (mpd_notifier_->dash_profile() == DashProfile::kLive) {
143  DCHECK(event_info_.empty());
144  // TODO(kqyang): Set mpd duration to |duration_seconds|, which is more
145  // accurate than the duration coded in the original media header.
146  if (mpd_notifier_->mpd_type() == MpdType::kStatic)
147  mpd_notifier_->Flush();
148  return;
149  }
150 
151  DCHECK(media_info_);
152  if (!internal::SetVodInformation(media_ranges, duration_seconds,
153  mpd_notifier_->use_segment_list(),
154  media_info_.get())) {
155  LOG(ERROR) << "Failed to generate VOD information from input.";
156  return;
157  }
158 
159  if (notification_id_) {
160  mpd_notifier_->NotifyMediaInfoUpdate(notification_id_.value(),
161  *media_info_);
162  } else {
163  if (!NotifyNewContainer())
164  return;
165  DCHECK(notification_id_);
166  }
167  // TODO(rkuroiwa): Use media_ranges.subsegment_ranges instead of caching the
168  // subsegments.
169  for (const auto& event_info : event_info_) {
170  switch (event_info.type) {
171  case EventInfoType::kSegment:
172  mpd_notifier_->NotifyNewSegment(
173  notification_id_.value(), event_info.segment_info.start_time,
174  event_info.segment_info.duration,
175  event_info.segment_info.segment_file_size);
176  break;
177  case EventInfoType::kKeyFrame:
178  // NO-OP for DASH.
179  break;
180  case EventInfoType::kCue:
181  mpd_notifier_->NotifyCueEvent(notification_id_.value(),
182  event_info.cue_event_info.timestamp);
183  break;
184  }
185  }
186  event_info_.clear();
187  mpd_notifier_->Flush();
188 }

190 void MpdNotifyMuxerListener::OnNewSegment(const std::string& file_name,
191  int64_t start_time,
192  int64_t duration,
193  uint64_t segment_file_size) {
194  if (mpd_notifier_->dash_profile() == DashProfile::kLive) {
195  mpd_notifier_->NotifyNewSegment(notification_id_.value(), start_time,
196  duration, segment_file_size);
197  if (mpd_notifier_->mpd_type() == MpdType::kDynamic)
198  mpd_notifier_->Flush();
199  } else {
200  EventInfo event_info;
201  event_info.type = EventInfoType::kSegment;
202  event_info.segment_info = {start_time, duration, segment_file_size};
203  event_info_.push_back(event_info);
204  }
