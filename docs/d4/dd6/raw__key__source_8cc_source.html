30 Status RawKeySource::GetKey(const std::string& stream_label,
31  EncryptionKey* key) {
32  DCHECK(key);
33  // Try to find the key with label |stream_label|. If it is not available,
34  // fall back to the default empty label if it is available.
35  auto iter = encryption_key_map_.find(stream_label);
36  if (iter == encryption_key_map_.end()) {
37  iter = encryption_key_map_.find(kEmptyDrmLabel);
38  if (iter == encryption_key_map_.end()) {
39  return Status(error::NOT_FOUND,
40  "Key for '" + stream_label + "' was not found.");
41  }
42  }
43  *key = *iter->second;
44  return Status::OK;
45 }

47 Status RawKeySource::GetKey(const std::vector<uint8_t>& key_id,
48  EncryptionKey* key) {
49  DCHECK(key);
50  for (const auto& pair : encryption_key_map_) {
51  if (pair.second->key_id == key_id) {
52  *key = *pair.second;
53  return Status::OK;
54  }
55  }
56  return Status(error::INTERNAL_ERROR,
57  "Key for key_id=" + base::HexEncode(&key_id[0], key_id.size()) +
58  " was not found.");
59 }

61 Status RawKeySource::GetCryptoPeriodKey(
62  uint32_t crypto_period_index,
63  int32_t crypto_period_duration_in_seconds,
64  const std::string& stream_label,
65  EncryptionKey* key) {
66  RETURN_IF_ERROR(GetKey(stream_label, key));
67 
68  // A naive key rotation algorithm is implemented here by left rotating the
69  // key, key_id and pssh. Note that this implementation is only intended for
70  // testing purpose. The actual key rotation algorithm can be much more
71  // complicated.
72  LOG(WARNING)
73  << "This naive key rotation algorithm should not be used in production.";
74  std::rotate(key->key_id.begin(),
75  key->key_id.begin() + (crypto_period_index % key->key_id.size()),
76  key->key_id.end());
77  std::rotate(key->key.begin(),
78  key->key.begin() + (crypto_period_index % key->key.size()),
79  key->key.end());
80  key->key_ids.clear();
81  key->key_ids.emplace_back(key->key_id);
82 
83  return Status::OK;
84 }

86 std::unique_ptr<RawKeySource> RawKeySource::Create(
87  const RawKeyParams& raw_key) {
88  std::vector<ProtectionSystemSpecificInfo> key_system_info;
89  if (!raw_key.pssh.empty()) {
90  if (!ProtectionSystemSpecificInfo::ParseBoxes(
91  raw_key.pssh.data(), raw_key.pssh.size(), &key_system_info)) {
92  LOG(ERROR) << "--pssh argument should be full PSSH boxes.";
93  return std::unique_ptr<RawKeySource>();
94  }
95  }
96 
97  std::vector<std::vector<uint8_t>> key_ids;
98  for (const auto& entry : raw_key.key_map)
99  key_ids.emplace_back(entry.second.key_id);
100 
101  EncryptionKeyMap encryption_key_map;
102  for (const auto& entry : raw_key.key_map) {
103  const std::string& drm_label = entry.first;
104  const RawKeyParams::KeyInfo& key_pair = entry.second;
105 
106  if (key_pair.key_id.size() != 16) {
107  LOG(ERROR) << "Invalid key ID size '" << key_pair.key_id.size()
108  << "', must be 16 bytes.";
109  return std::unique_ptr<RawKeySource>();
110  }
111  if (key_pair.key.size() != 16) {
112  // CENC only supports AES-128, i.e. 16 bytes.
113  LOG(ERROR) << "Invalid key size '" << key_pair.key.size()
114  << "', must be 16 bytes.";
115  return std::unique_ptr<RawKeySource>();
116  }
117  if (!key_pair.iv.empty() && key_pair.iv.size() != 8 && key_pair.iv.size() != 16) {
118  LOG(ERROR) << "Invalid IV '" << key_pair.iv.size()
119  << "', must be 8 or 16 bytes.";
120  return std::unique_ptr<RawKeySource>();
121  }
122 
123  std::unique_ptr<EncryptionKey> encryption_key(new EncryptionKey);
124  encryption_key->key_id = key_pair.key_id;
125  encryption_key->key_ids = key_ids;
126  encryption_key->key = key_pair.key;
127  encryption_key->iv = (key_pair.iv.empty()) ? raw_key.iv : key_pair.iv;
128  encryption_key->key_system_info = key_system_info;
129  encryption_key_map[drm_label] = std::move(encryption_key);
130  }
131 
132  return std::unique_ptr<RawKeySource>(
133  new RawKeySource(std::move(encryption_key_map)));
134 }
